144 STATIC void readEvalCIPs( void )
145 {

147  uint08 cip_cnt ;

152 
153  // offset tor first CIP in packet from start of buffer
154  soc = cmdPkt.soc ;
155 
156  for ( cip_cnt = 0 ; cip_cnt < MAX_CIP_COUNT ; cip_cnt++ ) {
157  // Normal return for valid packet
158  if ( soc == cmdPkt.eop ) {
159  return ;
160  }
161  // not enough data for cip header
162  if ( soc + CIP_HEADER_LEN > cmdPkt.eop ) {
163  errRaiseErr( 0, 0, 0 ) ;
164  return ;
165  }
166  // invalid sync
167  if ( Read4( cmdPkt.buf+soc ) != cip_sync ) {
168  errRaiseErr( 0, 0, 0 ) ;
169  return ;
170  }

172  // cip_type = Read1( cmdPkt.buf+soc+CIP_TYPE_OFFSET ) ;
173  cip_len = Read2( cmdPkt.buf+soc+4 ) ;
174  cip_type = Read1( cmdPkt.buf+soc+6 ) ;
175 
176  // missing data
177  if ( soc + cip_len > cmdPkt.eop ) {
178  errRaiseErr( 0, 0, 0 ) ;
179  return ;
180  }
181  // Other CIP types are not S/C -> Payload or are undefined.
182  // If processing a CIP raises an error then log it and
183  // move on to the next CIP
184 
185  data_start = soc + CIP_HEADER_LEN ;
186  data_len = cip_len - CIP_HEADER_LEN ;
187

189  case CIP_TIME_UPDATE: // time update
190  State.cmd_received++ ;
191  // exec_sclk_update( data_len, cmdPkt.buf + data_start ) ;
192  break ;
193  case CIP_INST_CMD : // instrument command
194  State.cmd_received++ ;
195  execExecOcams( cmdPkt.buf + data_start, data_len ) ;
196  break ;
197  case CIP_NOOP : // NOP
198  State.cmd_received++ ;
199  break ;
200  default:
201  // invalid CIP type
202  errRaiseErr( 0, 0, 0 ) ;
203  break ;
204  }

206  soc = soc + cip_len ;
207  }
