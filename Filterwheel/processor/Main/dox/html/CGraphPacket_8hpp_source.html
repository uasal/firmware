104  virtual bool IsValid(const uint8_t* Buffer, const size_t BufferCount, const size_t PacketStartPos, const size_t PacketEndPos) const
105  {
106  if ((PacketStartPos + sizeof(CGraphPacketHeader) + sizeof(CGraphPacketFooter)) > BufferCount) { return(false); }
107  const CGraphPacketHeader* Header = reinterpret_cast<const CGraphPacketHeader*>(&(Buffer[PacketStartPos]));
108  if ((PacketStartPos + sizeof(CGraphPacketHeader) + Header->PayloadLen + sizeof(CGraphPacketFooter)) > BufferCount) { return(false); }
109  if (CGraphMagikPacketStartToken != Header->PacketStartToken) { return(false); }
110  const CGraphPacketFooter* Footer = reinterpret_cast<const CGraphPacketFooter*>(&(Buffer[PacketStartPos + sizeof(CGraphPacketHeader) + Header->PayloadLen]));
111  if (CGraphMagikPacketEndToken != Footer->PacketEndToken) { return(false); }
112  uint32_t CRC = CRC32((uint8_t*)Header, sizeof(CGraphPacketHeader) + Header->PayloadLen);
113  if (CRC != Footer->CRC32) { return(false); }
114  return(true);
115  }

117  virtual size_t MakePacket(uint8_t* Buffer, const size_t BufferCount, const void* Payload, const uint16_t PayloadType, const size_t PayloadLen) const
118  {
119  if ( (NULL == Buffer) || ((NULL == Payload) && (0 != PayloadLen)) || (BufferCount < (sizeof(CGraphPacketHeader) + PayloadLen + sizeof(CGraphPacketFooter))) ) { return(0); }
120  CGraphPacketHeader Header;
121  Header.PayloadType = PayloadType;
122  Header.PayloadLen = PayloadLen;
123  memcpy(Buffer, &Header, sizeof(CGraphPacketHeader));
124  if (NULL != Payload) { memcpy(&(Buffer[sizeof(CGraphPacketHeader)]), Payload, PayloadLen); }
125  CGraphPacketFooter* Footer = reinterpret_cast<CGraphPacketFooter*>(&(Buffer[sizeof(CGraphPacketHeader) + PayloadLen]));
126  uint32_t CRC = CRC32(Buffer, sizeof(CGraphPacketHeader) + PayloadLen);
127  Footer->CRC32 = CRC;
128  Footer->PacketEndToken = CGraphMagikPacketEndToken;
129  return(sizeof(CGraphPacketHeader) + PayloadLen + sizeof(CGraphPacketFooter));
130  }
