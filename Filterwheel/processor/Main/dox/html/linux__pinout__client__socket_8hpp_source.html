63  virtual int init(const int HostPort, const char* HostName)
64  {
65  struct hostent* pHostInfo; /* holds info about a machine */
66  struct sockaddr_in Address; /* Internet socket address stuct */
67  long nHostAddress;
68  char strHostName[HOST_NAME_SIZE];
69  int nHostPort = 20339;
70 
71  if (-1 != hSocket) { deinit(); }
72 
73  strcpy(strHostName, "localhost");
74 
75  if (NULL != HostName)
76  {
77  strncpy(strHostName, HostName, HOST_NAME_SIZE);
78  strHostName[HOST_NAME_SIZE] = '\0';
79  }
80  if (0 != HostPort)
81  {
82  nHostPort = HostPort;
83  }
84 
85  #ifdef WIN32
86  formatf("\nStarting Winsock");
87  WSADATA wsaData;
88  if (WSAStartup(MAKEWORD(2,0), &wsaData) != 0)
89  {
90  printf("\nZigSockServer: WSAStartup failed.\n");
91  return 0;
92  }
93  #endif
94 
95  formatf("\nlinux_pinout_client_socket::init(): Making a socket");
96  /* make a socket */
97  hSocket=socket(AF_INET,SOCK_STREAM,IPPROTO_TCP);
98 
99  if(hSocket == -1)
100  {
101  #ifdef WIN32
102  formatf("\nlinux_pinout_client_socket::init(): Could not make a socket: %d\n", WSAGetLastError());
103  #else
104  perror("\nlinux_pinout_client_socket::init(): Could not make a socket\n");
105  #endif
106  return(errno);
107  }
108  else { formatf("\nlinux_pinout_client_socket::init(): Socket handle: %d\n", hSocket); }
109 
110  /* get IP address from name */
111  pHostInfo=gethostbyname(strHostName);
112  if (NULL == pHostInfo)
113  {
114  #ifdef WIN32
115  formatf("\nlinux_pinout_client_socket::init(): Could not gethostbyname(): %d\n", WSAGetLastError());
116  #else
117  perror("\nlinux_pinout_client_socket::init(): Could not gethostbyname(): ");
118  #endif
119  return(errno);
120  }
121  /* copy address into long */
122  memcpy(&nHostAddress,pHostInfo->h_addr,pHostInfo->h_length);
123 
124  /* fill address struct */
125  Address.sin_addr.s_addr=nHostAddress;
126  Address.sin_port=htons(nHostPort);
127  Address.sin_family=AF_INET;
128 
129  formatf("\nlinux_pinout_client_socket::init(): Connecting to %s", strHostName);
130  formatf(" on port %d", nHostPort);
131 
132  /* connect to host */
133  if(connect(hSocket,(struct sockaddr*)&Address,sizeof(Address))
134  == -1)
135  {
136  #ifdef WIN32
137  formatf("\nlinux_pinout_client_socket::init(): Could not connect to host: %d\n", WSAGetLastError());
138  #else
139  perror("\nlinux_pinout_client_socket::init(): Could not connect to host: ");
140  #endif
141  return(errno);
142  }
143  else { formatf("\nlinux_pinout_client_socket::init(): Connected.\n"); }
144 
145  return(IUartOK);
146  }

177  if (result < 0) { } //"You have an error"

180  if (FD_ISSET(hSocket, &sockset)) // The socket has data. For good measure, it's not a bad idea to test further

184  }
185  return(false);
