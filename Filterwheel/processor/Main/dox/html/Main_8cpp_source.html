79 int16_t ValidatedFWHomeStep()
80 {
81  int16_t HomeStep = 0;
82 
83  CGraphFWPositionStepRegister HomeA = FW->PosDetHomeA;
84  CGraphFWPositionStepRegister HomeB = FW->PosDetHomeB;
85 
86  formatf("\nESC-FW: ValidatedFWHomeStep precorrected: ");
87  HomeA.formatf();
88  formatf(";");
89  HomeB.formatf();
90 
91  bool AValid = (HomeA.OnStep != 0) && (HomeA.OffStep != 0);
92  bool BValid = (HomeB.OnStep != 0) && (HomeB.OffStep != 0);
93  //Seen some pretty weird values in all instances...but when they're weird, they match...
94  if (HomeA.OnStep == HomeA.OffStep) { AValid = false; }
95  if (HomeB.OnStep == HomeB.OffStep) { BValid = false; }
96 
97  //Correct for full revolutions:
98  HomeA.OnStep %= MotorFullCircleSteps;
99  HomeA.OffStep %= MotorFullCircleSteps;
100  HomeB.OnStep %= MotorFullCircleSteps;
101  HomeB.OffStep %= MotorFullCircleSteps;
102 
103  //Correct negatives
104  if (HomeA.OnStep < 0) { HomeA.OnStep += MotorFullCircleSteps; }
105  if (HomeA.OffStep < 0) { HomeA.OffStep += MotorFullCircleSteps; }
106  if (HomeB.OnStep < 0) { HomeB.OnStep += MotorFullCircleSteps; }
107  if (HomeB.OffStep < 0) { HomeB.OffStep += MotorFullCircleSteps; }
108 
109  //Correct splits
110  if (abs((int)HomeA.OnStep - (int)HomeA.OffStep) > (MotorFullCircleSteps / 2)) { (HomeA.OnStep < HomeA.OffStep) ? HomeA.OnStep += MotorFullCircleSteps : HomeA.OffStep += MotorFullCircleSteps; }
111  if (abs((int)HomeB.OnStep - (int)HomeB.OffStep) > (MotorFullCircleSteps / 2)) { (HomeB.OnStep < HomeB.OffStep) ? HomeB.OnStep += MotorFullCircleSteps : HomeB.OffStep += MotorFullCircleSteps; }
112 
113  //did that fail?
114  if (abs((int)HomeA.OnStep - (int)HomeA.OffStep) > (MotorFullCircleSteps / 2)) { AValid = false; }
115  if (abs((int)HomeB.OnStep - (int)HomeB.OffStep) > (MotorFullCircleSteps / 2)) { BValid = false; }
116 
117  formatf("\nESC-FW: ValidatedFWHomeStep post-corrected: ");
118  HomeA.formatf();
119  formatf(";");
120  HomeB.formatf();
121 
122  if (AValid) { HomeStep = HomeA.MidStep(); }
123  if (BValid) { HomeStep += HomeB.MidStep(); }
124  if (AValid && BValid) { HomeStep /= 2; }
125 
126  return(HomeStep);
127 }

129 void FWHome()
130 {
131  size_t i = 0;
132 
133  //Turn things on
134  CGraphFWHardwareControlRegister HCR;
135  HCR.PosLedsEnA = 1;
136  HCR.PosLedsEnB = 1;
137  HCR.MotorEnable = 1;
138  FW->ControlRegister = HCR;
139 
140  //stabilize
141  delayms(100);
142 
143  //Try to start with motor homed
144  CGraphFWMotorControlStatusRegister MCSR;
145  MCSR = FW->MotorControlStatus;
146  MCSR.SeekStep = 0;
147  FW->MotorControlStatus = MCSR;
148 
149  //Wait for it to move
150  for (i = 0; i < MotorFindHomeTimeoutMs; i++)
151  {
152  MCSR = FW->MotorControlStatus;
153  if (MCSR.SeekStep == MCSR.CurrentStep) { break; }
154  delayms(1);
155  }
156 
157  //Clear step registers
158  HCR.ResetSteps = 1;
159  FW->ControlRegister = HCR;
160  HCR.ResetSteps = 0;
161  FW->ControlRegister = HCR;
162 
163  //Now go around
164  MCSR.SeekStep = MotorFindHomeSteps;
165  FW->MotorControlStatus = MCSR;
166 
167  //Wait for it to move
168  for (i = 0; i < MotorFindHomeTimeoutMs; i++)
169  {
170  MCSR = FW->MotorControlStatus;
171  if (MCSR.SeekStep == MCSR.CurrentStep) { break; }
172  delayms(1);
173  }
174 
175  //Now send it home.
176  int16_t HomeStep = ValidatedFWHomeStep();
177 
178  //Backlash
179  MCSR.SeekStep = HomeStep - 45;
180  FW->MotorControlStatus = MCSR;
181 
182  formatf("\nESC-FW: Attempting to move motor to: %d.", HomeStep - 45);
183 
184  //Wait for it to move
185  for (i = 0; i < MotorFindHomeTimeoutMs; i++)
186  {
187  MCSR = FW->MotorControlStatus;
188  if ((HomeStep - 45) == MCSR.CurrentStep) { break; }
189  delayms(1);
190  }
191 
192  //Final resting place
193  MCSR.SeekStep = HomeStep;
194  FW->MotorControlStatus = MCSR;
195 
196  formatf("\nESC-FW: Attempting to move motor to: %d.", HomeStep);
197 
198  //Wait for it to move
199  for (i = 0; i < MotorFindHomeTimeoutMs; i++)
200  {
201  MCSR = FW->MotorControlStatus;
202  if (HomeStep == MCSR.CurrentStep) { break; }
203  delayms(1);
204  }
205 
206  formatf("\nESC-FW: Move complete; motor reports: %d, i = %u.", MCSR.CurrentStep, i);
207 
208  //Clear step registers (by definition home [filter #1] is now zero)
209  HCR.ResetSteps = 1;
210  FW->ControlRegister = HCR;
211  HCR.ResetSteps = 0;
212  FW->ControlRegister = HCR;
213 
214  CGraphFWPositionStepRegister HomeA = FW->PosDetHomeA;
215  CGraphFWPositionStepRegister HomeB = FW->PosDetHomeB;
216 
217  formatf("\nESC-FW: Step registers reset: ");
218  HomeA.formatf();
219  formatf(";");
220  HomeB.formatf();
221 
222  FWPosition = 1;
223 
224  //Turn things off
225  HCR.PosLedsEnA = 0;
226  HCR.PosLedsEnB = 0;
227  HCR.MotorEnable = 0;
228  FW->ControlRegister = HCR;
229 }
