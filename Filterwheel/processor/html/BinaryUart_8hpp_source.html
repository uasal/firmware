426 virtual void TxBinaryPacket(const uint16_t PayloadType, const uint32_t SerialNumber, const void* PayloadData, const size_t PayloadLen) const
427 {
428 uint8_t TxBuffer[TxBufferLenBytes];
429 size_t PktLen = Packet.MakePacket(TxBuffer, TxBufferLenBytes, PayloadData, PayloadType, PayloadLen);
430
431 // Transmit each byte of the packet through the UART pinout
432 for (size_t i = 0; i < PktLen; i++) { Pinout.putcqq(TxBuffer[i]); }
433
434 // Debug output: log the packet type, length, and contents in hex format
435 if (debug)
436 {
437 ::formatf("\n\nBinary Uart: Sending packet(%u, %u): ", PayloadType, PayloadLen);
438 for(size_t i = 0; i < PktLen; i++) { printf("%.2X:", TxBuffer[i]); }
439 printf("\n\n");
440 }
441 }
