31template<typename T, size_t list_size=16> class fixedlist
32{
33 private:

38 public:

45 fixedlist(const size_t len, const T t[]) // :
46 {
47 size_t max = len;
48 if (max > list_size) { max = list_size; }

162 for(size_t i = 0; i < list_size; i++)

165 {
