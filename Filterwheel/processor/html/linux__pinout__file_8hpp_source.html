95 {
96 char c = 0;
97
98 if (NULL != fd)
99 {
100 int numbytes = fread(&c, 1, 1, fd);
101 if (1 != numbytes)
102 {
103 printf("\nlinux_pinout_file::getcqq(): read failed with %d.\n", numbytes);
104 deinit();
105 }
106 }
107 else
108 {
109 printf("\nlinux_pinout_file::getcqq(): read on uninitialized file; please open file!\n");
110 }
111
112 return(c);
113 }

123 deinit();
124 }
125 }
126 else
127 {
128 printf("\nlinux_pinout_file::putcqq(): write on uninitialized file; please open file!\n");
129 }
130
131 return (c);
132 }
