315 int32_t ConvertOnceDiff(uint8_t chanpos, uint8_t channeg)
316 {
317 //Set fixedchannel mode:
318 ads1258details::config0register config0(1, 1, 1, 0, 0, 1);
319 config0.muxmod = 1;
320 WriteRegister(ads1258details::register_config0, config0.all);
321
322 //Set channel to read (always differential + & -)
323 ads1258details::muxschregister muxsch(chanpos, channeg);
324 WriteRegister(ads1258details::register_muxsch, muxsch.all);
325 uint8_t temp = 0;
326 ReadRegister(ads1258details::register_muxsch, temp);
327 if (muxsch.all != temp)
328 {
329 #ifdef DEBUGADC
330 formatf("ads1258::ConvertOnceDiff(): muxsch mismatch (reads:0x%.2X)\n", temp);
331 #endif
332 return(ads1258details::register_muxdif);
333 }
334
335 //Convert:
336 SendCommand(ads1258details::ads1258cmdheader(ads1258details::cmdtype_pulseconvert, false, 0));
337
338 //Wait
339 delayus(1000000);
340
341 //Read data & return:
342 return(ReadLastChannel(chanpos));
343 }

353 uint32_t GetScanChannelsMask()
354 {
355 uint32_t Mask = 0;
356
357 Mask |= ScanChansDiff;
358 Mask |= (ScanChansSELo << 8);
359 Mask |= (ScanChansSEHi << 16);
360 Mask |= (ScanChansInternal << 24);

365 void SetScanChannelsMask(const uint32_t Mask)
366 {
367 ScanChansDiff = Mask & 0xFF;
368 ScanChansSELo = (Mask >> 8) & 0xFF;
369 ScanChansSEHi = (Mask >> 16) & 0xFF;
370 ScanChansInternal = (Mask >> 24) & 0xFF;

374 {
375 if (chan >= ads1258details::ads1258numchannels) { return; }

392 bool IsScanChannel(const unsigned int chan)

399 unsigned int NumScanChannels()
400 {
401 unsigned int NumChannels = 0;
402
403 uint32_t ScanChannels = GetScanChannelsMask();
404
405 for (unsigned int i = 0; i < ads1258details::ads1258numchannels; i++)
406 {
407 if (ScanChannels & 1UL) { NumChannels++; }
408
409 ScanChannels = ScanChannels >> 1;
410 }
411
412 return(NumChannels);

510 virtual uint8_t Init(uint8_t gpio_output_mask = gpio_output_mask_all_outputs) //All outputs is reccommended in datasheet as inputs are not allowed to float for some ass reason.
511 {
512 //~ uint8_t Return = InitOK;
513 //see pg 41 of ads1258 revD datasheet.
514
515 //1. Reset spi - disable for 4096xFclk
516 {
517 enable(false);
518 delayus(10000);
519 }
520
521 //2. stop the converter: set "start" pin low
522 //<unimplemented in hardware, not strictly neccessary>
523
524 //3. Reset the converter
525 {
526 SendCommand(ads1258details::ads1258cmdheader(ads1258details::cmdtype_reset, false, 0));
527 delayus(1000);
528 }
529
530 //4. Config registers
531 //ads1258details::config0register config0(1, 0, 1, 0, 0, 1); //Chopper off
532 ads1258details::config0register config0(1, 0, 1, 0, 1, 1); //Chopper on
533 ads1258details::config1register config1(0, 7, 0, 0);
534 ads1258details::sysreadregister sysread(1, 1, 1, 1, 1);
535 {
536 WriteRegister(ads1258details::register_config0, config0.all);
537 WriteRegister(ads1258details::register_config1, config1.all);
538 WriteRegister(ads1258details::register_sysread, sysread.all);
539
540 //pg 42: set no-connected gpio's to outputs
541 WriteRegister(ads1258details::register_gpioc, gpio_output_mask); //all outputs
542 WriteRegister(ads1258details::register_gpiod, 0x00); //all zero
543 }
544
545 //5. Readback registers
546 {
547 uint8_t temp;
548
549 ReadRegister(ads1258details::register_config0, temp);
550 if (config0.all != temp)
551 {
552 //~ #ifdef DEBUGADC
553 ::formatf("ads1258::init(): config0 mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, config0.all);
554 //~ #endif
555 return(ads1258details::register_config0);
556 //~ Return |= ads1258details::register_config0;
557 }
558
559 ReadRegister(ads1258details::register_config1, temp);
560 if (config1.all != temp)
561 {
562 //~ #ifdef DEBUGADC
563 ::formatf("ads1258::init(): config1 mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, config1.all);
564 //~ #endif
565 return(ads1258details::register_config1);
566 //~ Return |= ads1258details::register_config1;
567 }
568
569 ReadRegister(ads1258details::register_sysread, temp);
570 if (sysread.all != temp)
571 {
572 //~ #ifdef DEBUGADC
573 ::formatf("ads1258::init(): sysread mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, sysread.all);
574 //~ #endif
575 return(ads1258details::register_sysread);
576 //~ Return |= ads1258details::register_sysread;
577 }
578
579 //pg 42: set no-connected gpio's to outputs
580 ReadRegister(ads1258details::register_gpioc, temp);
581 if (gpio_output_mask != temp)
582 {
583 //~ #ifdef DEBUGADC
584 ::formatf("ads1258::init(): gpioc mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, gpio_output_mask);
585 //~ #endif
586 return(ads1258details::register_gpioc);
587 //~ Return |= ads1258details::register_gpioc;
588 }
589
590 //(Don't read gpiod - we write 0x00, but if pins are set as inputs, they will not read 0x00)
591
592 //Check and see if the ID register matched 0x8B:
593 ReadRegister(ads1258details::register_idnum, temp);
594 if (0x8B != temp)
595 {
596 //~ #ifdef DEBUGADC
597 ::formatf("ads1258::init(): idnum mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, 0x8B);
598 //~ #endif
599 return(ads1258details::register_idnum);
600 //~ Return |= ads1258details::register_idnum;
601 }
602 }
603
604 return(InitOK);
605 //~ return(Return);
606 }
