41  void delayus(const unsigned long microseconds) //2^32 usec is ~1.5hrs; don't know why we'd delay longer than that...
42  {
43  struct timespec delay, remaining;
44  memset((char *)&delay, 0, sizeof(struct timespec));
45  memset((char *)&remaining, 0, sizeof(struct timespec));
46 
47  delay.tv_sec = microseconds / 1000000UL;
48  delay.tv_nsec = (microseconds - (delay.tv_sec * 1000000)) * 1000U; //values greater than 999999999 may cause errno == EINVAL
