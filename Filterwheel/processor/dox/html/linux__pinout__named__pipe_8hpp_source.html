110  struct timeval nowait;
111  memset((char *)&nowait,0,sizeof(nowait));
112  if (-1 == fd) { return(false); } //open?
113  FD_ZERO(&pipeset);
114  FD_SET(fd, &pipeset);
115  int result = select(fd + 1, &pipeset, NULL, NULL, &nowait);
116  if (result < 0)

118  //~ if (!silent) { printf("\nlinux_pinout_named_pipe::dataready(): select() failed with %ld.\n", (long int)errno); }
119  if (!silent) { perror("\nlinux_pinout_named_pipe::dataready(): select() failed"); }
120  }
121  else if (result == 1)
122  {
123  if (FD_ISSET(fd, &pipeset)) // The pipeet has data. For good measure, it's not a bad idea to test further

127  }
128  return(false);
