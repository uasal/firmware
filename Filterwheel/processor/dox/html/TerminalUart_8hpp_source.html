343  int maybegetline()

348 
349  if (RxCount >= rxbufferlenbytes) { RxCount = rxbufferlenbytes - 1; }
350 
351  //EOL (\r\n or \n)
352  if ('\n' == ch)
353  {
354  RxBuffer[RxCount] = '\0'; //terminate the string correctly
355  return(RxCount);
356  }
357 
358  //Backsp/del - remove a character
359  if ( (CHAR_BAK == ch) || (CHAR_DEL == ch) )
360  {
361  //just delete the last character
362  --RxCount;
363  if (RxCount >= rxbufferlenbytes) { RxCount = 0; }
364  RxBuffer[RxCount] = '\0'; //terminate the string correctly
365  return(0);
366  }
367 
368  //Esc - delete the current line
369  if (CHAR_ESC == ch)
370  {
371  //just delete the line
372  RxCount = 0;
373  RxBuffer[RxCount] = '\0'; //terminate the string correctly
374  return(0);
375  }
376 
377  //tab - re-input the last line
378  if (CHAR_TAB == ch) {
379 
380  RxCount = strnlen(RxBuffer, rxbufferlenbytes);
381 
382  //get rid of eol
383  if ('\n' == RxBuffer[RxCount]) { RxCount--; }
384  if ('\r' == RxBuffer[RxCount]) { RxCount--; }
385 
386  if (RxCount >= rxbufferlenbytes) { RxCount = rxbufferlenbytes - 1; }
387 
388  RxBuffer[RxCount] = '\0'; //terminate the string correctly
389 
390  //let the user see what they wrote
391  for (size_t i = 0; i < RxCount; i++) { Pinout.putcqq(RxBuffer[i]); }
392 
393  return(0); //so the user can edit the line
394  }
395 
396  RxBuffer[RxCount] = ch; //move to the next char in the buffer
397  RxCount++;
398 
399  //taking care of when the buffer overflows by just throwing chars away - if this is undesirable, set blockontxbufferfull() to true.
400  if (RxCount >= rxbufferlenbytes) { RxCount = rxbufferlenbytes - 1; }
401 
402  return(0);
