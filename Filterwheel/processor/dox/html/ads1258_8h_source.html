315  int32_t ConvertOnceDiff(uint8_t chanpos, uint8_t channeg)
316  {
317  //Set fixedchannel mode:
318  ads1258details::config0register config0(1, 1, 1, 0, 0, 1);

321

325  uint8_t temp = 0;
326  ReadRegister(ads1258details::register_muxsch, temp);

337

340 
341  //Read data & return:
342  return(ReadLastChannel(chanpos));

509  virtual uint8_t Init(uint8_t gpio_output_mask = gpio_output_mask_all_outputs) //All outputs is reccommended in datasheet as inputs are not allowed to float for some ass reason.
510  {
511  //~ uint8_t Return = InitOK;
512  //see pg 41 of ads1258 revD datasheet.
513 
514  //1. Reset spi - disable for 4096xFclk
515  {
516  enable(false);
517  delayus(10000);
518  }
519 
520  //2. stop the converter: set "start" pin low
521  //<unimplemented in hardware, not strictly neccessary>
522 
523  //3. Reset the converter
524  {
525  SendCommand(ads1258details::ads1258cmdheader(ads1258details::cmdtype_reset, false, 0));
526  delayus(1000);
527  }
528 
529  //4. Config registers
530  //ads1258details::config0register config0(1, 0, 1, 0, 0, 1); //Chopper off
531  ads1258details::config0register config0(1, 0, 1, 0, 1, 1); //Chopper on
532  ads1258details::config1register config1(0, 7, 0, 0);
533  ads1258details::sysreadregister sysread(1, 1, 1, 1, 1);
534  {
535  WriteRegister(ads1258details::register_config0, config0.all);
536  WriteRegister(ads1258details::register_config1, config1.all);
537  WriteRegister(ads1258details::register_sysread, sysread.all);
538 
539  //pg 42: set no-connected gpio's to outputs
540  WriteRegister(ads1258details::register_gpioc, gpio_output_mask); //all outputs
541  WriteRegister(ads1258details::register_gpiod, 0x00); //all zero
542  }
543 
544  //5. Readback registers
545  {
546  uint8_t temp;
547 
548  ReadRegister(ads1258details::register_config0, temp);
549  if (config0.all != temp)
550  {
551  //~ #ifdef DEBUGADC
552  ::formatf("ads1258::init(): config0 mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, config0.all);
553  //~ #endif
554  return(ads1258details::register_config0);
555  //~ Return |= ads1258details::register_config0;
556  }
557 
558  ReadRegister(ads1258details::register_config1, temp);
559  if (config1.all != temp)
560  {
561  //~ #ifdef DEBUGADC
562  ::formatf("ads1258::init(): config1 mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, config1.all);
563  //~ #endif
564  return(ads1258details::register_config1);
565  //~ Return |= ads1258details::register_config1;
566  }
567 
568  ReadRegister(ads1258details::register_sysread, temp);
569  if (sysread.all != temp)
570  {
571  //~ #ifdef DEBUGADC
572  ::formatf("ads1258::init(): sysread mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, sysread.all);
573  //~ #endif
574  return(ads1258details::register_sysread);
575  //~ Return |= ads1258details::register_sysread;
576  }
577 
578  //pg 42: set no-connected gpio's to outputs
579  ReadRegister(ads1258details::register_gpioc, temp);
580  if (gpio_output_mask != temp)
581  {
582  //~ #ifdef DEBUGADC
583  ::formatf("ads1258::init(): gpioc mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, gpio_output_mask);
584  //~ #endif
585  return(ads1258details::register_gpioc);
586  //~ Return |= ads1258details::register_gpioc;
587  }
588 
589  //(Don't read gpiod - we write 0x00, but if pins are set as inputs, they will not read 0x00)
590 
591  //Check and see if the ID register matched 0x8B:
592  ReadRegister(ads1258details::register_idnum, temp);
593  if (0x8B != temp)
594  {
595  //~ #ifdef DEBUGADC
596  ::formatf("ads1258::init(): idnum mismatch (reads:0x%.2X, wanted:0x%.2X)\n", temp, 0x8B);
597  //~ #endif
598  return(ads1258details::register_idnum);
599  //~ Return |= ads1258details::register_idnum;
600  }
601  }
602 
603  return(InitOK);
604  //~ return(Return);
605  }
