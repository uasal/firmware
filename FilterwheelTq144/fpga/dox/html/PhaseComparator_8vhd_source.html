55  signal InA_i : std_logic;

59 
60  --Constants of the right length...
61  constant One : std_logic_vector(MAX_CLOCK_BITS_DELTA - 1 downto 0) := std_logic_vector(to_unsigned(1, MAX_CLOCK_BITS_DELTA));
62  constant Zero : std_logic_vector(MAX_CLOCK_BITS_DELTA - 1 downto 0) := std_logic_vector(to_unsigned(0, MAX_CLOCK_BITS_DELTA));
63  --~ constant DeltaSaturated : std_logic_vector(MAX_CLOCK_BITS_DELTA - 1 downto 0) := std_logic_vector(to_unsigned((2 ** (MAX_CLOCK_BITS_DELTA - 1)) - 1, MAX_CLOCK_BITS_DELTA));
64  constant DeltaSaturated : std_logic_vector(MAX_CLOCK_BITS_DELTA - 1 downto 0) := (others => '1');
65 
66 begin
67 
68  IBUF_A : IBufP3Ports
69  port map (
70  clk => clk,
71  I => InA,
72  O => InA_i
73  );
74 
75  IBUF_B : IBufP3Ports
76  port map (
77  clk => clk,
78  I => InB,
79  O => InB_i
80  );
81 
82  -- Master clock drives most logic
83  process (clk, rst)
84  begin
85  if (rst = '1') then
86 
87  Delta_i <= Zero;
88  Delta <= Zero;
89 
90  else
91 
92  if ( (clk'event) and (clk = '1') ) then
93 
94  --Reset Counters on start of cycle (this is preferred to be both low, then if one clock is lost, delta will swing wildly, instead of stick at zero)
95  if ( (InA_i = '0') and (InB_i = '0') ) then
96 
97  Delta_i <= Zero;
98 
99  DeltaLatched <= '0';
100 
101  end if;
102 
103  --If A preceeds B, we count down from 0
104  if ( (InA_i = '1') and (InB_i = '0') ) then
105 
106  --saturate:
107  if (Delta_i < DeltaSaturated) then
108 
109  Delta_i <= Delta_i + One;
110 
111  end if;
112 
113  end if;
114 
115  --If B preceeds A, we count up from 0
116  if ( (InA_i = '0') and (InB_i = '1') ) then
117 
118  --saturate:
119  if ( (Delta_i > (DeltaSaturated + One)) or (Delta_i = Zero) ) then --signed logic in unsigned var =>big numbers are negative
120 
121  Delta_i <= Delta_i - One;
122 
123  end if;
124 
125  end if;
126 
127  --If they're both end of cycle, sit on our hands...
128  if ( (InA_i = '1') and (InB_i = '1') ) then
129 
130  if (DeltaLatched = '0') then
131 
132  Delta <= Delta_i;
133 
134  DeltaLatched <= '1';
135 
136  end if;
137 
138  end if;
139 
140  end if;
141 
142  end if;
143 
144  end process; --(clock)
145 
146 end PhaseComparatorr;
