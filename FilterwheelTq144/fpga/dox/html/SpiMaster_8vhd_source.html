38 entity SpiMasterPorts is
39  generic
40  (
41  CLOCK_DIVIDER : natural := 1000; --how much do you want to knock down the global clock to get to the spi clock rate?
42  BYTE_WIDTH : natural := 1; --how many bytes per transaction?
43  CPOL : std_logic := '0'; --'standard' spi knob - inverts clock polarity (0 seems to be the standard, 1 less common)
44  CPHA : std_logic := '0'--; --'standard' spi knob - inverts clock phase (0 seems to be the standard, 1 less common)
45  );
46  port
47  (
48  --Globals
49  clk : in std_logic;
50  rst : in std_logic; --we do a single spi message for every reset cycle, reset must toggle for more than a single transfer.
51

56 
57  --Registers
58  DataToMosi : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);

60  XferComplete : out std_logic--;
61  );
62 end SpiMasterPorts;
63 
64 architecture SpiMaster of SpiMasterPorts is
65 
66  signal SpiBitPos : natural range 0 to (BYTE_WIDTH * 8); --Which clock cycle are we on anyway?
67 
68  signal ClkDiv : natural range 0 to ((CLOCK_DIVIDER / 2) - 1); --Hold the clock divider chain
69 
70  signal Sck_i : std_logic;
71  signal Mosi_i : std_logic;
72 
73  signal DataToMosi_i : std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0); --register input data
74  signal DataToMosiLatched : std_logic;
75 
76  signal XferComplete_i : std_logic;
77 
78 begin
79 
80  --~ Sck <= Sck_i xor CPOL; --Allow for Sck to be inverted
81  Sck <= Sck_i; --Allow for Sck to be inverted
82  Mosi <= Mosi_i;
83  XferComplete <= XferComplete_i;
84 
85  process (clk, rst, Miso, DataToMosi)
86  begin
87 
88  -- Chip select is reset, and starts transfer when it clocks to '0'.
89  if (rst = '1') then
90 
91  Sck_i <= not(CPOL);
92  Mosi_i <= DataToMosi((BYTE_WIDTH * 8) - 1); --get the first bit out there asap
93  DataToMosi_i <= std_logic_vector(to_unsigned(0, BYTE_WIDTH * 8));
94  DataToMosiLatched <= '0';
95  DataFromMiso((BYTE_WIDTH * 8) - 1) <= Miso; --grab the first bit asap
96  DataFromMiso((BYTE_WIDTH * 8) - 2 downto 0) <= std_logic_vector(to_unsigned(0, (BYTE_WIDTH * 8) - 1));
97  XferComplete_i <= '0';
98  SpiBitPos <= (BYTE_WIDTH * 8); --MSB first transfers; for LSB first, load "000" instead.
99  ClkDiv <= 0;
100 
101  else
102 
103  if ( ( clk'event) and (clk = '1') ) then
104 
105  --Run latch
106  if (DataToMosiLatched = '0') then
107 
108  DataToMosi_i <= DataToMosi;
109  DataToMosiLatched <= '1';
110 
111  end if;
112 
113  --Run clock divider
114  if (ClkDiv < ((CLOCK_DIVIDER / 2) - 1)) then --Since we flop sck back & forth, run divider twice as fast...
115 
116  ClkDiv <= ClkDiv + 1;
117  if (SpiBitPos = (BYTE_WIDTH * 8)) then Mosi_i <= DataToMosi((BYTE_WIDTH * 8) - 1); end if; --still time to update the MSB for Mosi. GZHOU
118 
119 
120  --Run bus
121  else
122 
123  ClkDiv <= 0;
124 
125  --Done?
126  if (XferComplete_i = '0') then
127 
128 
129  if (Sck_i = ((not(CPOL)) xor CPHA)) then --transition mosi when SCK != CPOL
130 
131  if (SpiBitPos > 0) then
132 
133  Mosi_i <= DataToMosi_i(SpiBitPos - 1);
134 
135  end if;
136 
137  --Here is the transition to the final state where we idle until the next reset sequence
138  if (SpiBitPos = 0) then
139 
140  XferComplete_i <= '1';
141 
142  end if;
143 
144  else --read miso when SCK = CPOL
145 
146  DataFromMiso(SpiBitPos - 1) <= Miso;
147 
148  --Move to next bit
149  if (SpiBitPos > 0) then
150 
151  SpiBitPos <= SpiBitPos - 1;
152 
153  end if;
154 
155  end if; --Sck_i = not(CPOL)
156 
157  if (SpiBitPos /= 0) then
158 
159  --Toggle spi bus clock output on every divider rollover
160  Sck_i <= not(Sck_i);
161 
162  end if;
163 
164  end if; --XferComplete_i = '0'
165 
166  end if; --ClkDiv
167 
168  end if; --clk
169 
170  end if; --rst
171 
172  end process; --(Sck, rst)
173 
174 end SpiMaster;
