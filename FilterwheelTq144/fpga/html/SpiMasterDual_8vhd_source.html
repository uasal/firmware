33library IEEE;
34use IEEE.STD_LOGIC_1164.ALL;
35use IEEE.STD_LOGIC_UNSIGNED.ALL;
36use IEEE.NUMERIC_STD.all;
37
38entity SpiMasterDualPorts is
39 generic
40 (
41 CLOCK_DIVIDER : natural := 1000; --how much do you want to knock down the global clock to get to the spi clock rate?
42 BYTE_WIDTH : natural := 1; --how many bytes per transaction?
43 CPOL : std_logic := '0'; --'standard' spi knob - inverts clock polarity (0 seems to be the standard, 1 less common)
44 CPHA : std_logic := '0'--; --'standard' spi knob - inverts clock phase (0 seems to be the standard, 1 less common)
45 );
46 port
47 (
48 --Globals
49 clk : in std_logic;
50 rst : in std_logic; --we do a single spi message for every reset cycle, reset must toggle for more than a single transfer.
51
52 --Spi pins
53 MosiA : out std_logic;
54 MosiB : out std_logic;
55 Sck : out std_logic;
56 MisoA : in std_logic;
57 MisoB : in std_logic;
58
59 --Registers
60 DataToMosiA : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
61 DataToMosiB : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
62 DataFromMisoA : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
63 DataFromMisoB : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
64 XferComplete : out std_logic--;
65 );
66end SpiMasterDualPorts;
67
68architecture SpiMasterDual of SpiMasterDualPorts is
69
70 signal SpiBitPos : natural range 0 to (BYTE_WIDTH * 8); --Which clock cycle are we on anyway?
71
72 signal ClkDiv : natural range 0 to ((CLOCK_DIVIDER / 2) - 1); --Hold the clock divider chain
73
74 signal Sck_i : std_logic;
75 signal MosiA_i : std_logic;
76 signal MosiB_i : std_logic;
77
78 signal DataToMosiA_i : std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0); --register input data
79 signal DataToMosiB_i : std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0); --register input data
80 signal DataToMosiLatched : std_logic;
81
82 signal XferComplete_i : std_logic;
83
84begin
85
86 --~ Sck <= Sck_i xor CPOL; --Allow for Sck to be inverted
87 Sck <= Sck_i; --Allow for Sck to be inverted
88 MosiA <= MosiA_i;
89 MosiB <= MosiB_i;
90 XferComplete <= XferComplete_i;
91
92 process (clk, rst, MisoA, MisoB, DataToMosiA, DataToMosiB)
93 begin
94
95 -- Chip select is reset, and starts transfer when it clocks to '0'.
96 if (rst = '1') then
97
98 Sck_i <= not(CPOL);
99 MosiA_i <= DataToMosiA((BYTE_WIDTH * 8) - 1); --get the first bit out there asap
100 MosiB_i <= DataToMosiB((BYTE_WIDTH * 8) - 1); --get the first bit out there asap
101 DataToMosiA_i <= std_logic_vector(to_unsigned(0, BYTE_WIDTH * 8));
102 DataToMosiB_i <= std_logic_vector(to_unsigned(0, BYTE_WIDTH * 8));
103 DataToMosiLatched <= '0';
104 DataFromMisoA((BYTE_WIDTH * 8) - 1) <= MisoA; --grab the first bit asap
105 DataFromMisoA((BYTE_WIDTH * 8) - 2 downto 0) <= std_logic_vector(to_unsigned(0, (BYTE_WIDTH * 8) - 1));
106 DataFromMisoB((BYTE_WIDTH * 8) - 1) <= MisoB; --grab the first bit asap
107 DataFromMisoB((BYTE_WIDTH * 8) - 2 downto 0) <= std_logic_vector(to_unsigned(0, (BYTE_WIDTH * 8) - 1));
108 XferComplete_i <= '0';
109 SpiBitPos <= (BYTE_WIDTH * 8); --MSB first transfers; for LSB first, load "000" instead.
110 ClkDiv <= 0;
111
112 else
113
114 if ( ( clk'event) and (clk = '1') ) then
115
116 --Run latch
117 if (DataToMosiLatched = '0') then
118
119 DataToMosiA_i <= DataToMosiA;
120 DataToMosiB_i <= DataToMosiB;
121
122 DataToMosiLatched <= '1';
123
124 end if;
125
126 --Run clock divider
127 if (ClkDiv < ((CLOCK_DIVIDER / 2) - 1)) then --Since we flop sck back & forth, run divider twice as fast...
128
129 ClkDiv <= ClkDiv + 1;
130 if (SpiBitPos = (BYTE_WIDTH * 8)) then
131
132 MosiA_i <= DataToMosiA((BYTE_WIDTH * 8) - 1);
133 MosiB_i <= DataToMosiB((BYTE_WIDTH * 8) - 1);
134
135 end if; --still time to update the MSB for Mosi. GZHOU
136
137
138 --Run bus
139 else
140
141 ClkDiv <= 0;
142
143 --Done?
144 if (XferComplete_i = '0') then
145
146
147 if (Sck_i = ((not(CPOL)) xor CPHA)) then --transition mosi when SCK != CPOL
148
149 if (SpiBitPos > 0) then
150
151 MosiA_i <= DataToMosiA_i(SpiBitPos - 1);
152 MosiB_i <= DataToMosiB_i(SpiBitPos - 1);
153
154 end if;
155
156 --Here is the transition to the final state where we idle until the next reset sequence
157 if (SpiBitPos = 0) then
158
159 XferComplete_i <= '1';
160
161 end if;
162
163 else --read miso when SCK = CPOL
164
165 DataFromMisoA(SpiBitPos - 1) <= MisoA;
166 DataFromMisoB(SpiBitPos - 1) <= MisoB;
167
168 --Move to next bit
169 if (SpiBitPos > 0) then
170
171 SpiBitPos <= SpiBitPos - 1;
172
173 end if;
174
175 end if; --Sck_i = not(CPOL)
176
177 if (SpiBitPos /= 0) then
178
179 --Toggle spi bus clock output on every divider rollover
180 Sck_i <= not(Sck_i);
181
182 end if;
183
184 end if; --XferComplete_i = '0'
185
186 end if; --ClkDiv
187
188 end if; --clk
189
190 end if; --rst
191
192 end process; --(Sck, rst)
193
194end SpiMasterDual;
