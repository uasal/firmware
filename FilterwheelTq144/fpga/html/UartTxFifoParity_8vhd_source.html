146begin
147
148 ------------------------------------------ Atomic Clock Uart+Fifo ---------------------------------------------------
149
150 --~ zigfiforeset has clock domain issues, UartTxfiforeset probably does too. async->sync reset?
151 --~ so does uclk, pretty much everything in the uart domain
152 --~ fifos are run by uclk, but read by clk via datamapper...
153 --~ should run write & data out of slow domain, run fifo from master.
154 --~ also need to add useatomicclk flag to fpgacontrol register
155
156 UartTxFifo : gated_fifo
157 generic map
158 (
159 WIDTH_BITS => 8,
160 DEPTH_BITS => FIFO_BITS--,
161 )
162 port map
163 (
164 clk => clk,
165 rst => rst,
166 wone_i => WriteStrobe,
167 data_i => WriteData,
168 rone_i => ReadStrobe,
169 full_o => FifoFull,
170 empty_o => FifoEmpty_i,
171 data_o => OutgoingTxByte,
172 count_o => FifoCount,
173 r_ack => FifoReadAck--,
174 );
175 FifoEmpty <= FifoEmpty_i;
176
177 BitClockDiv : ClockDividerPorts
178 generic map
179 (
180 CLOCK_DIVIDER => natural( (real(UART_CLOCK_FREQHZ) / (real(BAUDRATE)) ) + 0.5)
181 )
182 port map
183 (
184 clk => uclk,
185 rst => rst,
186 div => BitClock
187 );
188
189 BitClockOut <= BitClock;
190
191 IBufStartTx : IBufP2Ports --cross the clk domain
192 port map
193 (
194 clk => BitClock,
195 I => StartTx,
196 O => StartTx_i--,
197 );
198
199 UartTxUart : UartTxParity
200 generic map
201 (
202 PARITY_EVEN => PARITY_EVEN
203 )
204 port map
205 (
206 clk => BitClock,
207 reset => rst,
208 Go => StartTx_i,
209 TxD => Txd,
210 Busy => TxInProgress_i_i,
211 Data => OutgoingTxByte
212 );
213
214 IBufTxInProgress_i : IBufP2Ports --cross the clk domain
215 port map
216 (
217 clk => clk,
218 I => TxInProgress_i_i,
219 O => TxInProgress_i--,
220 );
221
222 TxInProgress <= TxInProgress_i;
223
224 IBufCts : IBufP2Ports --cross the clk domain
225 port map
226 (
227 clk => clk,
228 I => Cts,
229 O => Cts_i--,
230 );
231
232
233 --Move data from fifo out uart as it's available...
234 process (clk, rst, CurrentState)
235 begin
236
237 if (rst = '1') then
238
239 CurrentState <= Idle;
240 NextState <= Idle;
241 ReadStrobe <= '0';
242 StartTx <= '0';
243
244 else
245
246 if ( (clk'event) and (clk = '1') ) then
247
248 CurrentState <= NextState;
249
250 case CurrentState is
251
252 when Idle =>
253
254 ReadStrobe <= '0';
255
256 StartTx <= '0';
257
258 --~ if (FifoEmpty_i = '0') then
259
260 if ( (FifoEmpty_i = '0') and (TxInProgress_i = '0') and (Cts_i = '0') ) then
261
262 NextState <= StartRead;
263
264 end if;
265
266 when StartRead =>
267
268 if (FifoReadAck = '0') then
269
270 ReadStrobe <= '1';
271
272 NextState <= WaitAck;
273
274 end if;
275
276 when WaitAck =>
277
278 if (FifoReadAck = '1') then
279
280 ReadStrobe <= '0';
281
282 StartTx <= '1';
283
284 end if;
285
286 if (TxInProgress_i = '1') then
287
288 NextState <= Tx;
289
290 end if;
291
292 when Tx =>
293
294 StartTx <= '0';
295
296 if (TxInProgress_i = '0') then
297
298 NextState <= Idle;
299
300 end if;
301
302 when others => -- ought never to get here...
303
304 NextState <= Idle;
305
306 end case;
307
308 end if;
309
310 end if;
311
312 end process;
313
314end implementation;
