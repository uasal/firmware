88 component SpiMasterTrioPorts is
89 generic (
90 CLOCK_DIVIDER : integer := 4;
91 BYTE_WIDTH : natural := 1;
92 CPOL : std_logic := '0'--;
93 );
94 port
95 (
96 clk : in std_logic;
97 rst : in std_logic;
98 MosiA : out std_logic;
99 MosiB : out std_logic;
100 MosiC : out std_logic;
101 Sck : out std_logic;
102 MisoA : in std_logic;
103 MisoB : in std_logic;

107 DataToMosiC : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
108 DataFromMisoA : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
109 DataFromMisoB : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
110 DataFromMisoC : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);

114
115 signal SpiRst : std_logic; --kicks off / inhibits transfer of a sample out of the A/D
116 signal SpiXferComplete : std_logic;
117 signal LastSpiXferComplete : std_logic;
118
119 --~ signal DacClk : std_logic;
120 signal LastWriteDac : std_logic;
121 signal DacWriteOutA_i : std_logic_vector(23 downto 0);
122 signal DacWriteOutB_i : std_logic_vector(23 downto 0);
123 signal DacWriteOutC_i : std_logic_vector(23 downto 0);
124 signal DacReadbackA_i : std_logic_vector(23 downto 0);
125 signal DacReadbackB_i : std_logic_vector(23 downto 0);
126 signal DacReadbackC_i : std_logic_vector(23 downto 0);
127
128begin
129
130 Spi : SpiMasterTrioPorts
131 generic map (
132 --~ CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 4,
133 --~ CLOCK_DIVIDER => 100, --1MHz
134 CLOCK_DIVIDER => 10, --10MHz (50MHz max MAX5719)
135 --~ CLOCK_DIVIDER => 4, --25MHz
136 --~ CLOCK_DIVIDER => 3, --33MHz
137 --~ CLOCK_DIVIDER => 2, --50MHz
138 BYTE_WIDTH => 3,
139 CPOL => '0'--, --'inverted' SCK polarity?
140 )
141 port map
142 (
143 clk => clk, --runs off the same clock as the A/D to keep everything nicely aligned & quiet
144 rst => SpiRst, --every sample requires a set/rst sequence to run spimaster
145 MosiA => MosiA,
146 MosiB => MosiB,
147 MosiC => MosiC,
148 Sck => Sck,
149 MisoA => MisoA,
150 MisoB => MisoB,
151 MisoC => MisoC,
152 DataToMosiA => DacWriteOutA_i,
153 DataToMosiB => DacWriteOutB_i,
154 DataToMosiC => DacWriteOutC_i,
155 DataFromMisoA => DacReadbackA_i,
156 DataFromMisoB => DacReadbackB_i,
157 DataFromMisoC => DacReadbackC_i,
158 XferComplete => SpiXferComplete--,
159 );
160
161 nCsA <= SpiRst; --these concepts are synchronous in this design
162 nCsB <= SpiRst; --these concepts are synchronous in this design
163 nCsC <= SpiRst; --these concepts are synchronous in this design
164
165 TransferComplete <= SpiRst; --these concepts are synchronous in this design
166
167 --Read A/D:
168 process (clk, rst, WriteDac, SpiXferComplete)
169 begin
170
171 if (rst = '1') then
172
173 SpiRst <= '1';
174 LastWriteDac <= '0';
175 LastSpiXferComplete <= '0';
176 DacWriteOutA_i <= x"000000";
177 DacWriteOutB_i <= x"000000";
178 DacWriteOutC_i <= x"000000";
179
180 else
181
182 if ( (clk'event) and (clk = '1') ) then
183
184 --Follow Drdy
185 if (WriteDac /= LastWriteDac) then
186
187 LastWriteDac <= WriteDac;
188
189 --Here we go...
190 if (WriteDac = '1') then
191
192 --Latch inputs & initiate reading the data.
193 DacWriteOutA_i <= DacWriteOutA;
194 DacWriteOutB_i <= DacWriteOutB;
195 DacWriteOutC_i <= DacWriteOutC;
196 SpiRst <= '0';
197
198 end if;
199
200 else
201
202 --Wait for Spi xfer to complete, then grab the sample and we're done
203 if (SpiXferComplete /= LastSpiXferComplete) then
204
205 LastSpiXferComplete <= SpiXferComplete;
206
207 if (SpiXferComplete = '1') then
208
209 --Grab read back
210 DacReadbackA <= DacReadbackA_i;
211 DacReadbackB <= DacReadbackB_i;
212 DacReadbackC <= DacReadbackC_i;
213
214 --turn off spi master bus
215 SpiRst <= '1';
216
217 end if;
218
219 end if;
220
221 end if;
222
223 end if;
224
225 end if;
226
227 end process;
228
229end SpiDacTrio;
