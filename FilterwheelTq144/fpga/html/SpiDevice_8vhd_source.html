44entity SpiDevicePorts is
45 generic (
46 CLOCK_DIVIDER : natural := 4;
47 BIT_WIDTH : natural := 8;
48 CPOL : std_logic := '0'; --'standard' spi knob - inverts clock polarity (0 seems to be the standard, 1 less common)
49 CPHA : std_logic := '0'--; --'standard' spi knob - inverts clock phase (0 seems to be the standard, 1 less common)
50 );
51 port (
52
53 --Globals
54 clk : in std_logic;

56

62
63 --Control signals
64 WriteOut : in std_logic_vector(BIT_WIDTH - 1 downto 0);

67 TransferComplete : out std_logic--;

82 component SpiMasterPorts is
83 generic (
84 CLOCK_DIVIDER : integer := 4;
85 BYTE_WIDTH : natural := 1;
86 CPOL : std_logic := '0'; --'standard' spi knob - inverts clock polarity (0 seems to be the standard, 1 less common)
87 CPHA : std_logic := '0'--; --'standard' spi knob - inverts clock phase (0 seems to be the standard, 1 less common)
88 );
89 port
90 (
91 clk : in std_logic;
92 rst : in std_logic;
93 Mosi : out std_logic;
94 Sck : out std_logic;
95 Miso : in std_logic;
96 DataToMosi : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
97 DataFromMiso : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
98 XferComplete : out std_logic--;
99 );
100 end component;
101
102 signal SpiRst : std_logic; --kicks off / inhibits transfer of a sample out of the A/D
103 signal SpiXferComplete : std_logic;
104 signal LastSpiXferComplete : std_logic;
105
106 --~ signal DacClk : std_logic;
107 signal LastTransfer : std_logic;
108 signal TransferActuallyComplete : std_logic; --When we're done & we reset the spi bus it's version of this signal goes back to zero
109 signal Readback_i : std_logic_vector(BIT_WIDTH - 1 downto 0);
110
111begin
112
113 Spi : SpiMasterPorts
114 generic map (
115 CLOCK_DIVIDER => CLOCK_DIVIDER, --100kHz
116 BYTE_WIDTH => BIT_WIDTH / 8 ,
117 CPOL => CPOL, --'inverted' SCK polarity?
118 CPHA => CPHA--, --'inverted' SCK phase?
119 )
120 port map
121 (
122 clk => clk, --runs off the same clock as the A/D to keep everything nicely aligned & quiet
123 rst => SpiRst, --every sample requires a set/rst sequence to run spimaster
124 Mosi => Mosi, --we don't actually send anything to the A/D, all it needs is the sample trigger/clock
125 Sck => Sck,
126 Miso => Miso,
127 DataToMosi => WriteOut, --we don't actually send anything to the A/D
128 DataFromMiso => Readback_i,
129 XferComplete => SpiXferComplete--,
130 );
131
132 nCs <= SpiRst; --these concepts are synchronous in this design
133
134 TransferComplete <= TransferActuallyComplete;
135
136 --~ TransferComplete <= SpiRst; --these concepts are synchronous in this design
137
138 --Read A/D:
139 process (clk, rst, Transfer, SpiXferComplete)
140 begin
141
142 if (rst = '1') then
143
144 SpiRst <= '1';
145 LastTransfer <= '0';
146 LastSpiXferComplete <= '0';
147 TransferActuallyComplete <= '0';
148
149 else
150
151 if ( (clk'event) and (clk = '1') ) then
152
153 --Follow Drdy
154 if (Transfer /= LastTransfer) then
155
156 LastTransfer <= Transfer;
157
158 --Here we go...
159 if (Transfer = '1') then
160
161 --Initiate reading the data.
162 SpiRst <= '0';
163
164 TransferActuallyComplete <= '0';
165
166 end if;
167
168 else
169
170 --Wait for Spi xfer to complete, then grab the sample and we're done
171 if (SpiXferComplete /= LastSpiXferComplete) then
172
173 LastSpiXferComplete <= SpiXferComplete;
174
175 if (SpiXferComplete = '1') then
176
177 --Grab read back
178 Readback <= Readback_i;
179
180 TransferActuallyComplete <= '1';
181
182 --turn off spi master bus
183 SpiRst <= '1';
184
185 end if;
186
187 end if;
188
189 end if;
190
191 end if;
192
193 end if;
194
195 end process;
196
197end SpiDevice;
