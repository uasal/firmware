95begin

97 Spi : SpiMasterPorts
98 generic map (
99 --~ CLOCK_DIVIDER => (MASTER_CLOCK_FREQHZ / 1000000) + 1, --ltc244x datahseet specs sclk <= 20MHz, so let's try 1MHz
100 CLOCK_DIVIDER => (MASTER_CLOCK_FREQHZ / 100000) + 1, --ltc244x datahseet specs sclk <= 20MHz, so let's try 1MHz
101 BYTE_WIDTH => 4,
102 CPOL => '0'--,
103 )
104 port map
105 (
106 clk => clk,
107 rst => SpiRst,
108 Mosi => Mosi,
109 Sck => Sck,
110 Miso => Miso,
111 DataToMosi => DataToMosi, --We just tie Mosi to zero (or optionally 0xFF) to use continuous direct reads on nDrdy
112 DataFromMiso => DataFromMiso,
113 XferComplete => SpiXferComplete--,
114 );
115
116 nCs <= SpiRst; --Just run nCs low while we're doing a transfer...
117
118 TimestampReq <= TimestampReq_i;
119
120 Sample <= Sample_i;
121
122 process (clk, rst, nDrdy, SpiXferComplete)
123 begin
124
125 --here's how to invert a signal:
126 --~ for i in ChannelInverted'range loop
127 --~ ChannelInverted(ChannelInverted'left - i) <= NextChannel(i);
128 --~ end loop;
129
130 if (rst = '1') then
131
132 LastnDrdy <= NoDataReady; --otherwise we'll never start up, because drdy will start at zero and never change, so we'll never toggle cs and start a new cycle...
133 SpiRst <= '1';
134 Sample_i <= ltc244xsampleinit;
135 Config_i <= ltc244xconfiginit;
136 SampleLatched <= '0';
137 TimestampReq_i <= '0';
138
139 else
140
141 if ( (clk'event) and (clk = '1') ) then
142
143 --Follow Drdy
144 if (nDrdy /= LastnDrdy) then
145
146 LastnDrdy <= nDrdy;
147
148 if (nDrdy = DataReady) then
149
150 --Set outgoing config (i.e. channel)
151 --~ Config_i.Channel <= "10010"; --DEBUG! Ch4 (se) is IAdj
152 Config_i.Channel <= NextChannel;
153 Config_i.DataRate <= NextDataRate;
154 Config_i.DoubleSpeedAndOneCycleLatency <= NextDoubleRate;
155 DataToMosi <= ltc244xconfig_to_std_logic(Config_i);
156
157 --Just read the data...
158 SpiRst <= '0';
159 SampleLatched <= '0';
160
161 --Ask for a timestamp
162 TimestampReq_i <= '1';
163
164 end if;
165
166 end if;
167
168 --Make sure TimestampReq is a single clock:
169 if (TimestampReq_i = '1') then TimestampReq_i <= '0'; end if;
170
171 --Wait for Spi xfer to complete, then grab the Sample_i...
172 if (SpiXferComplete = '1') then
173
174 Sample_i <= std_logic_to_ltc244xsample(DataFromMiso);
175 --~ Sample_i <= std_logic_to_ltc244xsample(x"12345678"); --DEBUG!
176 SampleLatched <= '1';
177 SpiRst <= '1';
178
179 end if;
180
181 end if;
182
183 end if;
184
185 end process;
186
187end ltc244x_i;
