130 ------------------------------------------ Atomic Clock Uart+Fifo ---------------------------------------------------
131
132 --~ zigfiforeset has clock domain issues, UartTxfiforeset probably does too. async->sync reset?
133 --~ so does uclk, pretty much everything in the uart domain
134 --~ fifos are run by uclk, but read by clk via datamapper...
135 --~ should run write & data out of slow domain, run fifo from master.
136 --~ also need to add useatomicclk flag to fpgacontrol register
137
138 UartTxFifo : gated_fifo
139 generic map
140 (
141 WIDTH_BITS => 8,
142 DEPTH_BITS => FIFO_BITS--,
143 )
144 port map
145 (
146 clk => clk,
147 rst => rst,
148 wone_i => WriteStrobe,
149 data_i => WriteData,
150 rone_i => ReadStrobe,
151 full_o => FifoFull,
152 empty_o => FifoEmpty_i,
153 data_o => OutgoingTxByte,
154 count_o => FifoCount,
155 r_ack => FifoReadAck--,
156 );
157 FifoEmpty <= FifoEmpty_i;
158
159 BitClock <= uclk;
160 BitClockOut <= BitClock;
161
162 IBufStartTx : IBufP2Ports --cross the clk domain
163 port map
164 (
165 clk => BitClock,
166 I => StartTx,
167 O => StartTx_i--,
168 );
169
170 UartTxUart : UartTx
171 port map
172 (
173 clk => BitClock,
174 reset => rst,
175 Go => StartTx_i,
176 TxD => Txd,
177 Busy => TxInProgress_i_i,
178 BitCountOut => BitCountOut,
179 Data => OutgoingTxByte
180 );
181
182 IBufTxInProgress_i : IBufP2Ports --cross the clk domain
183 port map
184 (
185 clk => clk,
186 I => TxInProgress_i_i,
187 O => TxInProgress_i--,
188 );
189
190 TxInProgress <= TxInProgress_i;
191
192 IBufCts : IBufP2Ports --cross the clk domain
193 port map
194 (
195 clk => clk,
196 I => Cts,
197 O => Cts_i--,
198 );
199
200
201 --Move data from fifo out uart as it's available...
202 process (clk, rst, CurrentState)
203 begin
204
205 if (rst = '1') then
206
207 CurrentState <= Idle;
208 NextState <= Idle;
209 ReadStrobe <= '0';
210 StartTx <= '0';
211
212 else
213
214 if ( (clk'event) and (clk = '1') ) then
215
216 CurrentState <= NextState;
217
218 case CurrentState is
219
220 when Idle =>
221
222 ReadStrobe <= '0';
223
224 StartTx <= '0';
225
226 --~ if (FifoEmpty_i = '0') then
227
228 if ( (FifoEmpty_i = '0') and (TxInProgress_i = '0') and (Cts_i = '0') ) then
229
230 NextState <= StartRead;
231
232 end if;
233
234 when StartRead =>
235
236 if (FifoReadAck = '0') then
237
238 ReadStrobe <= '1';
239
240 if (TxInProgress_i = '0') then
241
242 NextState <= WaitAck;
243
244 end if;
245
246 end if;
247
248 when WaitAck =>
249
250 if (FifoReadAck = '1') then
251
252 ReadStrobe <= '0';
253
254 StartTx <= '1';
255
256 end if;
257
258 if (TxInProgress_i = '1') then
259
260 NextState <= Tx;
261
262 end if;
263
264 when Tx =>
265
266 StartTx <= '0';
267
268 if (TxInProgress_i = '0') then
269
270 NextState <= Idle;
271
272 end if;
273
274 when others => -- ought never to get here...
275
276 NextState <= Idle;
277
278 end case;
279
280 end if;
281
282 end if;
283
284 end process;
285
286end implementation;
