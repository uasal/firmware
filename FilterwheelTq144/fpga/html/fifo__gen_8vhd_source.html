40architecture rtl of fifo is
41 constant DEPTH : natural := 2**DEPTH_BITS;
42 -- Declare a RAM array data type
43 type ram_type is array (0 to DEPTH - 1) of std_logic_vector(WIDTH_BITS - 1 downto 0);
44 -- Shared variable to infer block ram
45 --~ shared variable RAM : ram_type := (others => (others => '0'));
46 shared variable RAM : ram_type;
47 -- Read/Write address pointers
48 signal raddr_r, waddr_r : unsigned(DEPTH_BITS - 1 downto 0) := (others => '0');
49 -- Async. counter change/Read/Write flag
50 signal do_count : std_logic := '0';
51 signal do_write : std_logic := '0';
52 signal do_read : std_logic := '0';
53 -- Fill counter
54 signal counter_r : natural range 0 to DEPTH := 0;
55 signal empty_r : std_logic := '1';
56 signal full_r : std_logic := '0';
57 signal data_r : std_logic_vector(WIDTH_BITS - 1 downto 0) := (others => '0');
58begin
59 do_read <= re_i and not empty_r;
60 do_write <= we_i and not full_r;
61 do_count <= '1' when do_read /= do_write else '0';
62 empty_o <= empty_r;
63 full_o <= full_r;
64 data_o <= data_r;
65
66 update: process(rst, clk)
67 begin
68 if rst = '1' then
69 counter_r <= 0;
70 raddr_r <= (others => '0');
71 waddr_r <= (others => '0');
72 full_r <= '0';
73 empty_r <= '1';
74 elsif rising_edge(clk) then
75
76 if (full_r = '0') then
77 count_o <= std_logic_vector(to_unsigned(counter_r, DEPTH_BITS));
78 else
79 count_o <= std_logic_vector(to_unsigned((2**DEPTH_BITS) - 1, DEPTH_BITS));
80 end if;
81
82 if counter_r = 0 or (counter_r = 1 and do_read = '1' and
83 do_write = '0') then
84 empty_r <= '1';
85 else
86 empty_r <= '0';
87 end if;
88
89 if counter_r = DEPTH or (counter_r = DEPTH - 1
90 and do_write = '1' and do_read = '0') then
91 full_r <= '1';
92 else
93 full_r <= '0';
94 end if;
95
96 if do_read = '1' then
97 raddr_r <= raddr_r + 1;
98 end if;
99
100 if do_write = '1' then
101 waddr_r <= waddr_r + 1;
102 end if;
103
104 if do_count = '1' then
105 if do_read = '1' then
106 counter_r <= counter_r - 1;
107 else
108 counter_r <= counter_r + 1;
109 end if;
110 end if;
111 end if;
112 end process update;
113
114 dpram_porta: process(clk, do_write)
115 begin
116 if rising_edge(clk) and do_write = '1' then
117 RAM(to_integer(waddr_r)) := data_i;
118 end if;
119 end process dpram_porta;
120
121 dpram_portb: process(clk, do_read)
122 begin
123 if rising_edge(clk) then
124 if do_read = '1' then
125 data_r <= RAM(to_integer(raddr_r));
126 r_ack <= '1';
127 else
128 r_ack <= '0';
129 end if;
130 end if;
131 end process dpram_portb;
132end rtl;
