59 --~ DacSetpointOut <= DacSetpoints(ReadAddressController, ReadAddressDac, ReadAddressChannel); --for asynchronous read

61 -- <where '%' is mod operator>
62 -- n = (z * numy * numx) + (y * numx) + x
63 -- z = n / (numy * numx)
64 -- y = (n / numx) % numy
65 -- x = n % numx
66
67 WriteAddressController <= WriteAddress / (DMActuatorsPerDac * DMMDacsPerControllerBoard);
68 WriteAddressDac <= (WriteAddress / DMActuatorsPerDac) mod DMMDacsPerControllerBoard;
69 WriteAddressChannel <= WriteAddress mod DMActuatorsPerDac;
70
71 process (clk, rst)
72 begin
73 if (rst = '1') then
74
75 DacSetpointOut <= x"000000"; --for synchronous read
76
77 else
78 if ( (clk'event) and (clk = '1') ) then
79
80 DacSetpointOut <= DacSetpoints(ReadAddressController, ReadAddressDac, ReadAddressChannel); --for synchronous read
81
82 if (WriteReq = '1') then
83
84 DacSetpoints(WriteAddressController, WriteAddressDac, WriteAddressChannel) := DacSetpointIn;
85
86 end if;
87
88 end if;
89
90 end if;
91 end process;

93end DmDacRam;
