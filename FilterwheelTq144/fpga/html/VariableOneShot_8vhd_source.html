44architecture VariableOneShot of VariableOneShotPorts is
45
46 signal ClkDiv : natural range 0 to (((2**WIDTH_BITS) / 2) - 1) := 0;
47 signal shot_i : std_logic := SHOT_RST_STATE;
48
49begin
50
51 shot <= shot_i;
52
53 process (clk, rst)
54 begin
55
56 if (rst = '1') then
57
58 ClkDiv <= 0;
59 shot_i <= SHOT_RST_STATE;
60
61 else
62
63 if ( (clk'event) and (clk = '1') ) then
64
65 --Generate the master DE-sampling clock
66 if (ClkDiv < delay_clks) then --/2 because we're only one one edge of master clock, not both, -1 because we count 0 too...
67
68 ClkDiv <= ClkDiv + 1;
69
70 shot_i <= SHOT_PRETRIGGER_STATE;
71
72 else
73
74 shot_i <= not(SHOT_PRETRIGGER_STATE);
75
76 end if;
77
78 end if;
79
80 end if;
81
82 end process;
83
84end VariableOneShot;
