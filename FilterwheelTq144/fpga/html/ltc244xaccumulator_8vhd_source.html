99 signal Retreived_ltc244xaccumulator : ltc244xaccumulator;

102 --state machine for incoming samples
103 type AccumSampleStates is ( WaitSample, GetLastAccum, ResetLastAccum, Accumulate, WriteAccumResult, WriteCompleteBumpChannel );
104 signal AccumSampleNextState : AccumSampleStates;
105 signal AccumSampleCurrentState : AccumSampleStates;

108 signal AdcSampleLatched : std_logic;
109 signal LastAdcSampleLatched : std_logic;

117 type ram_type is array (0 to DEPTH - 1) of std_logic_vector(63 downto 0);

121begin
122
123 --Debug signals
124 Dbg1 <= ResetAccum(0);
125 Dbg2 <= AdcSampleLatched;

127 ltc244x_i : ltc244xAutoscanPorts
128 generic map
129 (
130 MASTER_CLOCK_FREQHZ => MASTER_CLOCK_FREQHZ--,
131 )
132 port map
133 (
134 clk => clk,
135 rst => rst,
136 nDrdy => nDrdy,
137 nCs => nCs,
138 Sck => Sck,
139 Mosi => Mosi,
140 Miso => Miso,
141 NextChannel => NextChannel,
142 --~ NextChannel => "10000", --debug; always read Ch0 so we can figure out if we have hardware vs. software problems...
143 NextDatarate => LTC244X_DATARATE,
144 NextDoubleRate => LTC244X_DOUBLERATE,
145 Sample => AdcSample,
146 SampleLatched => AdcSampleLatched,
147 TimestampReq => open--,
148 );
149
150 --This process grabs samples from the a/d and accumulates them
151 process(clk, rst, AdcSampleLatched)
152 begin
153
154 if (rst = '1') then
155
156 LastAdcSampleLatched <= '0';
157 LastReadAdcSample <= '0';
158 InvalidStateReached <= '0';
159
160 Channel <= "00000";
161 NextChannel <= "00000";
162
163 Accumulated_ltc244xaccumulator <= ltc244xaccumulator_init;
164 Retreived_ltc244xaccumulator <= ltc244xaccumulator_init;
165 ResetAccum <= x"FFFFFFFF";
166
167 AccumSampleNextState <= WaitSample;
168 AccumSampleCurrentState <= WaitSample;
169
170 else
171
172 if ( (clk'event) and (clk = '1') ) then
173
174 --Did the uC want to read a sample?
175 if (ReadAdcSample /= LastReadAdcSample) then
176
177 --Capture edge
178 LastReadAdcSample <= ReadAdcSample;
179
180 --Latch sample on rising edge
181 if (ReadAdcSample = '1') then
182
183 AdcSampleToRead <= std_logic_to_ltc244xaccum(RAM(to_integer(unsigned(AdcChannelReadIndex))));
184
185 --Clear accumulator on falling edge
186 else
187
188 --Set reset flag
189 ResetAccum(to_integer(unsigned(AdcChannelReadIndex))) <= '1';
190
191 end if;
192
193 else
194
195 --Change state when requested
196 AccumSampleCurrentState <= AccumSampleNextState;
197
198 case AccumSampleCurrentState is
199
200 when WaitSample =>
201
202 --Wait state; don't change states until we get a new sample
203
204 --If we get here, we're in the intended state machine
205 InvalidStateReached <= '0';
206
207 --Do we have a new sample from the A/D?
208 if (AdcSampleLatched /= LastAdcSampleLatched) then
209
210 --Capture edge
211 LastAdcSampleLatched <= AdcSampleLatched;
212
213 if (AdcSampleLatched = '1') then
214
215 AccumSampleNextState <= GetLastAccum;
216
217 --Once we get a sample the channel is gonna get bumpped when we're done.
218 NextChannel <= Channel + "00001";
219
220 end if;
221
222 end if;
223
224 when GetLastAccum =>
225
226 --Single clock state
227 AccumSampleNextState <= ResetLastAccum;
228
229 --Get old accumulator
230 Retreived_ltc244xaccumulator <= std_logic_to_ltc244xaccum(RAM(to_integer(unsigned(Channel))));
231
232 when ResetLastAccum =>
233
234 --Single clock state
235 AccumSampleNextState <= Accumulate;
236
237 if ( ResetAccum(to_integer(unsigned(Channel))) = '1' ) then
238
239 --Reset NumAccums
240 Retreived_ltc244xaccumulator.NumAccums <= x"0000"; --Increment NumAccums
241
242 --Reset Accumulator
243 Retreived_ltc244xaccumulator.Sample <= x"000000000000";
244
245 --Clear reset flag
246 ResetAccum(to_integer(unsigned(Channel))) <= '0';
247
248 end if;
249
250 when Accumulate =>
251
252 --Single clock state
253 AccumSampleNextState <= WriteAccumResult;
254
255 --Saturate
256 if (Retreived_ltc244xaccumulator.NumAccums < x"FFFF") then
257
258 --Increment NumAccums
259 Accumulated_ltc244xaccumulator.NumAccums <= Retreived_ltc244xaccumulator.NumAccums + x"0001"; --Increment NumAccums
260
261 --Increment Accumulator
262 --~ Accumulated_ltc244xaccumulator.Sample <= Retreived_ltc244xaccumulator.Sample + AdcSample.Sample; --Accumulate
263
264 --Or
265
266 --Debug: don't actually accumulate until we get semi-realistic values from A/D...
267
268 --Just overwite accumulator with new sample
269 Accumulated_ltc244xaccumulator.Sample(28 downto 0) <= AdcSample.Sample;
270
271 --Sign extension - you only wanna do this for a bipolar supply...
272 --~ if (AdcSample.Sample(28) = '0') then
273 Accumulated_ltc244xaccumulator.Sample(47 downto 29) <= "0000000000000000000";
274 --~ else
275 --~ Accumulated_ltc244xaccumulator.Sample(47 downto 29) <= "1111111111111111111";
276 --~ end if;
277
278 --/Debug
279
280 end if;
281
282 when WriteAccumResult =>
283
284 --Single clock state
285 AccumSampleNextState <= WriteCompleteBumpChannel;
286
287 --Write to ram (if the ram isn't single-cycle we're in trouble)
288 RAM(to_integer(unsigned(Channel))) := ltc244xaccum_to_std_logic(Accumulated_ltc244xaccumulator);
289
290 when WriteCompleteBumpChannel =>
291
292 --Single clock state
293 AccumSampleNextState <= WaitSample;
294
295 --Bump Channel
296 Channel <= NextChannel;
297
298 when others => -- ought never to get here...reset everything!
299
300 InvalidStateReached <= '1';
301
302 LastAdcSampleLatched <= '0';
303 LastReadAdcSample <= '0';
304
305 Channel <= "00000";
306 NextChannel <= "00000";
307
308 Accumulated_ltc244xaccumulator <= ltc244xaccumulator_init;
309 Retreived_ltc244xaccumulator <= ltc244xaccumulator_init;
310 ResetAccum <= x"FFFFFFFF";
311
312 AccumSampleNextState <= WaitSample;
313 AccumSampleCurrentState <= WaitSample;
314
315 end case;
316
317 end if;
318
319 end if;
320
321 end if;
322
323 end process;
324
325end ltc244xaccumulator_i;
