55 --~ constant ClockDividerRollover : std_logic_vector(15 downto 0) := std_logic_vector(to_unsigned((CLOCK_FREQ / 1000) - 1, 16)); --51.2MHz / 51200 - 1 = 1kHz = 1 mS
56 constant ClockDividerRollover : natural := (CLOCK_FREQ / 1000) - 1; --51.2MHz / 51200 - 1 = 1kHz = 1 mS
57 --~ signal ClockDivider : std_logic_vector(15 downto 0);
58 signal ClockDivider : natural range 0 to ClockDividerRollover := 0;
59
60 signal Milliseconds_i : std_logic_vector(9 downto 0);

68begin
69

73 GeneratedPPS <= '1' when (MilliSeconds_i < 64) else '0';
74
75 process (rst, clk, PPS, SetTime, Milliseconds_i)
76 begin
77
78 if (rst = '1') then
79
80 ClockDivider <= 0;
81 MilliSeconds_i <= "0000000000";
82 Seconds_i <= "0000000000000000000000";
83 HavePPS <= '0';
84 LastPPS <= '1'; --pulled up by default in h/w, so must idle at 1; also makes it do both edges before acting on the first pps. Now, when the GPS pulls in, does it move a whole bunch and fubar our autosync?
85 TimeWasSet <= '0';
86
87 else
88
89 if ( (clk'event) and (clk = '1') ) then
90
91 --Edge of PPS is sacred: resets everything...
92 if ( (LastPPS /= PPS) and (GeneratePPS = '0') ) then --the generatepps flag keeps us from listening when we are the source of pps instead of gps!
93
94 LastPPS <= PPS;
95
96 HavePPS <= '1';
97
98 if (PPS = '1') then
99
100 Seconds_i <= Seconds_i + 1;
101
102 Milliseconds_i <= std_logic_vector(to_unsigned(0, 10));
103
104 ClockDivider <= 0;
105
106 end if;
107
108 else
109
110 if (ClockDivider < ClockDividerRollover) then
111
112 ClockDivider <= ClockDivider + 1; --wait for a millisecond to go by...
113
114 --Handle setting the time
115 if ( (SetTime = '1') and (TimeWasSet = '0') ) then
116
117 TimeWasSet <= '1';
118
119 if (Seconds_i /= SetTimeSeconds) then
120
121 SetChangedTime <= '1';
122
123 Seconds_i <= SetTimeSeconds;
124
125 else
126
127 SetChangedTime <= '0';
128
129 end if;
130
131 end if;
132
133 if (SetTime = '0') then TimeWasSet <= '0'; end if;
134
135 else --edge of next millisecond
136
137 ClockDivider <= 0;
138
139 if (Milliseconds_i >= std_logic_vector(to_unsigned(1000, 10))) then
140
141 --Trip this, next PPS sets it back.
142 HavePPS <= '0';
143
144 Seconds_i <= Seconds_i + 1;
145
146 Milliseconds_i <= std_logic_vector(to_unsigned(0, 10));
147
148 else
149
150 if (Milliseconds_i >= std_logic_vector(to_unsigned(999, 10))) then
151
152 if (HavePPS = '0') then
153
154 Seconds_i <= Seconds_i + 1;
155
156 Milliseconds_i <= std_logic_vector(to_unsigned(0, 10));
157
158 else
159
160 Milliseconds_i <= Milliseconds_i + 1;
161
162 end if;
163
164 else
165
166 Milliseconds_i <= Milliseconds_i + 1;
167
168 end if;
169
170 end if;
171
172 end if;
173
174 end if;
175
176 end if;
177
178 end if;
179
180 end process; --(clock)
181
182end RtcCounter;
