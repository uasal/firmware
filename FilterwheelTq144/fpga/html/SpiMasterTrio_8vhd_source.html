33library IEEE;
34use IEEE.STD_LOGIC_1164.ALL;
35use IEEE.STD_LOGIC_UNSIGNED.ALL;
36use IEEE.NUMERIC_STD.all;
37
38entity SpiMasterTrioPorts is
39 generic
40 (
41 CLOCK_DIVIDER : natural := 1000; --how much do you want to knock down the global clock to get to the spi clock rate?
42 BYTE_WIDTH : natural := 1; --how many bytes per transaction?
43 CPOL : std_logic := '0'; --'standard' spi knob - inverts clock polarity (0 seems to be the standard, 1 less common)
44 CPHA : std_logic := '0'--; --'standard' spi knob - inverts clock phase (0 seems to be the standard, 1 less common)
45 );
46 port
47 (
48 --Globals
49 clk : in std_logic;
50 rst : in std_logic; --we do a single spi message for every reset cycle, reset must toggle for more than a single transfer.
51
52 --Spi pins
53 MosiA : out std_logic;
54 MosiB : out std_logic;
55 MosiC : out std_logic;
56 Sck : out std_logic;
57 MisoA : in std_logic;
58 MisoB : in std_logic;
59 MisoC : in std_logic;
60
61 --Registers
62 DataToMosiA : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
63 DataToMosiB : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
64 DataToMosiC : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
65 DataFromMisoA : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
66 DataFromMisoB : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
67 DataFromMisoC : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
68 XferComplete : out std_logic--;
69 );
70end SpiMasterTrioPorts;
71
72architecture SpiMasterTrio of SpiMasterTrioPorts is
73
74 signal SpiBitPos : natural range 0 to (BYTE_WIDTH * 8); --Which clock cycle are we on anyway?
75
76 signal ClkDiv : natural range 0 to ((CLOCK_DIVIDER / 2) - 1); --Hold the clock divider chain
77
78 signal Sck_i : std_logic;
79 signal MosiA_i : std_logic;
80 signal MosiB_i : std_logic;
81 signal MosiC_i : std_logic;
82
83 signal DataToMosiA_i : std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0); --register input data
84 signal DataToMosiB_i : std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0); --register input data
85 signal DataToMosiC_i : std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0); --register input data
86 signal DataToMosiLatched : std_logic;
87
88 signal XferComplete_i : std_logic;
89
90begin
91
92 --~ Sck <= Sck_i xor CPOL; --Allow for Sck to be inverted
93 Sck <= Sck_i; --Allow for Sck to be inverted
94 MosiA <= MosiA_i;
95 MosiB <= MosiB_i;
96 MosiC <= MosiC_i;
97 XferComplete <= XferComplete_i;
98
99 process (clk, rst, MisoA, MisoB, MisoC, DataToMosiA, DataToMosiB, DataToMosiC)
100 begin
101
102 -- Chip select is reset, and starts transfer when it clocks to '0'.
103 if (rst = '1') then
104
105 Sck_i <= not(CPOL);
106 MosiA_i <= DataToMosiA((BYTE_WIDTH * 8) - 1); --get the first bit out there asap
107 MosiB_i <= DataToMosiB((BYTE_WIDTH * 8) - 1); --get the first bit out there asap
108 MosiC_i <= DataToMosiC((BYTE_WIDTH * 8) - 1); --get the first bit out there asap
109 DataToMosiA_i <= std_logic_vector(to_unsigned(0, BYTE_WIDTH * 8));
110 DataToMosiB_i <= std_logic_vector(to_unsigned(0, BYTE_WIDTH * 8));
111 DataToMosiC_i <= std_logic_vector(to_unsigned(0, BYTE_WIDTH * 8));
112 DataToMosiLatched <= '0';
113 DataFromMisoA((BYTE_WIDTH * 8) - 1) <= MisoA; --grab the first bit asap
114 DataFromMisoA((BYTE_WIDTH * 8) - 2 downto 0) <= std_logic_vector(to_unsigned(0, (BYTE_WIDTH * 8) - 1));
115 DataFromMisoB((BYTE_WIDTH * 8) - 1) <= MisoB; --grab the first bit asap
116 DataFromMisoB((BYTE_WIDTH * 8) - 2 downto 0) <= std_logic_vector(to_unsigned(0, (BYTE_WIDTH * 8) - 1));
117 DataFromMisoC((BYTE_WIDTH * 8) - 1) <= MisoC; --grab the first bit asap
118 DataFromMisoC((BYTE_WIDTH * 8) - 2 downto 0) <= std_logic_vector(to_unsigned(0, (BYTE_WIDTH * 8) - 1));
119 XferComplete_i <= '0';
120 SpiBitPos <= (BYTE_WIDTH * 8); --MSB first transfers; for LSB first, load "000" instead.
121 ClkDiv <= 0;
122
123 else
124
125 if ( ( clk'event) and (clk = '1') ) then
126
127 --Run latch
128 if (DataToMosiLatched = '0') then
129
130 DataToMosiA_i <= DataToMosiA;
131 DataToMosiB_i <= DataToMosiB;
132 DataToMosiC_i <= DataToMosiC;
133
134 DataToMosiLatched <= '1';
135
136 end if;
137
138 --Run clock divider
139 if (ClkDiv < ((CLOCK_DIVIDER / 2) - 1)) then --Since we flop sck back & forth, run divider twice as fast...
140
141 ClkDiv <= ClkDiv + 1;
142 if (SpiBitPos = (BYTE_WIDTH * 8)) then
143
144 MosiA_i <= DataToMosiA((BYTE_WIDTH * 8) - 1);
145 MosiB_i <= DataToMosiB((BYTE_WIDTH * 8) - 1);
146 MosiC_i <= DataToMosiC((BYTE_WIDTH * 8) - 1);
147
148 end if; --still time to update the MSB for Mosi. GZHOU
149
150
151 --Run bus
152 else
153
154 ClkDiv <= 0;
155
156 --Done?
157 if (XferComplete_i = '0') then
158
159
160 if (Sck_i = ((not(CPOL)) xor CPHA)) then --transition mosi when SCK != CPOL
161
162 if (SpiBitPos > 0) then
163
164 MosiA_i <= DataToMosiA_i(SpiBitPos - 1);
165 MosiB_i <= DataToMosiB_i(SpiBitPos - 1);
166 MosiC_i <= DataToMosiC_i(SpiBitPos - 1);
167
168 end if;
169
170 --Here is the transition to the final state where we idle until the next reset sequence
171 if (SpiBitPos = 0) then
172
173 XferComplete_i <= '1';
174
175 end if;
176
177 else --read miso when SCK = CPOL
178
179 DataFromMisoA(SpiBitPos - 1) <= MisoA;
180 DataFromMisoB(SpiBitPos - 1) <= MisoB;
181 DataFromMisoC(SpiBitPos - 1) <= MisoC;
182
183 --Move to next bit
184 if (SpiBitPos > 0) then
185
186 SpiBitPos <= SpiBitPos - 1;
187
188 end if;
189
190 end if; --Sck_i = not(CPOL)
191
192 if (SpiBitPos /= 0) then
193
194 --Toggle spi bus clock output on every divider rollover
195 Sck_i <= not(Sck_i);
196
197 end if;
198
199 end if; --XferComplete_i = '0'
200
201 end if; --ClkDiv
202
203 end if; --clk
204
205 end if; --rst
206
207 end process; --(Sck, rst)
208
209end SpiMasterTrio;
