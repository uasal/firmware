44architecture OneShot of OneShotPorts is
45
46 --~ constant CLOCK_DIVIDER : natural := ((natural(real(CLOCK_FREQHZ) * DELAY_SECONDS)) / 2) - 1;
47 constant CLOCK_DIVIDER : natural := ((natural(real(CLOCK_FREQHZ) * DELAY_SECONDS)) / 1) - 1;
48 signal ClkDiv : natural range 0 to CLOCK_DIVIDER := 0;
49 signal shot_i : std_logic := SHOT_RST_STATE;
50
51begin
52
53 shot <= shot_i;
54
55 process (clk, rst)
56 begin
57
58 if (rst = '1') then
59
60 ClkDiv <= 0;
61 shot_i <= SHOT_RST_STATE;
62
63 else
64
65 if ( (clk'event) and (clk = '1') ) then
66
67 --Generate the master DE-sampling clock
68 if (ClkDiv < CLOCK_DIVIDER) then --/2 because we're only one one edge of master clock, not both, -1 because we count 0 too...
69
70 ClkDiv <= ClkDiv + 1;
71
72 shot_i <= SHOT_PRETRIGGER_STATE;
73
74 else
75
76 shot_i <= not(SHOT_PRETRIGGER_STATE);
77
78 end if;
79
80 end if;
81
82 end if;
83
84 end process;
85
86end OneShot;
