80 begin
81

83 waddr_o <= waddr_R;

92

121 end if;
122 end if;
123
124 if do_write = '1' then
125 waddr_r <= waddr_r + 1;
126 end if;
127
128 if do_count = '1' then
129 if do_read = '1' then
130 counter_r <= counter_r - 1;
131 else

133 end if;

135 end if;

138 dpram_porta: process(clk, do_write)
139 begin
140 if rising_edge(clk) and do_write = '1' then
141 RAM(to_integer(waddr_r)) := data_i;
142 end if;
143 end process dpram_porta;

145 dpram_portb: process(clk, do_read)
146 begin
147 if rising_edge(clk) then

154 end if;
155 end if;
