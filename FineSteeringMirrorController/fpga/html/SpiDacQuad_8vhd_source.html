124 signal SpiRst : std_logic; --kicks off / inhibits transfer of a sample out of the A/D
125 signal SpiXferComplete : std_logic;
126 signal LastSpiXferComplete : std_logic;
127
128 --~ signal DacClk : std_logic;
129 signal LastWriteDac : std_logic;
130 signal DacWriteOutA_i : std_logic_vector(23 downto 0);
131 signal DacWriteOutB_i : std_logic_vector(23 downto 0);
132 signal DacWriteOutC_i : std_logic_vector(23 downto 0);

134 signal DacReadbackA_i : std_logic_vector(23 downto 0);
135 signal DacReadbackB_i : std_logic_vector(23 downto 0);

139begin
140
141 Spi : SpiMasterQuadPorts
142 generic map (
143 --~ CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 4,
144 --~ CLOCK_DIVIDER => 100, --1MHz
145 CLOCK_DIVIDER => 10, --10MHz (50MHz max MAX5719)
146 --~ CLOCK_DIVIDER => 4, --25MHz
147 --~ CLOCK_DIVIDER => 3, --33MHz
148 --~ CLOCK_DIVIDER => 2, --50MHz
149 BYTE_WIDTH => 3,
150 CPOL => '0'--, --'inverted' SCK polarity?
151 )
152 port map
153 (
154 clk => clk, --runs off the same clock as the A/D to keep everything nicely aligned & quiet
155 rst => SpiRst, --every sample requires a set/rst sequence to run spimaster
156 MosiA => MosiA,
157 MosiB => MosiB,
158 MosiC => MosiC,
159 MosiD => MosiD,
160 Sck => Sck,
161 MisoA => MisoA,
162 MisoB => MisoB,
163 MisoC => MisoC,
164 MisoD => MisoD,
165 DataToMosiA => DacWriteOutA_i,
166 DataToMosiB => DacWriteOutB_i,
167 DataToMosiC => DacWriteOutC_i,
168 DataToMosiD => DacWriteOutD_i,
169 DataFromMisoA => DacReadbackA_i,
170 DataFromMisoB => DacReadbackB_i,
171 DataFromMisoC => DacReadbackC_i,
172 DataFromMisoD => DacReadbackD_i,
173 XferComplete => SpiXferComplete--,
174 );
175
176 nCsA <= SpiRst; --these concepts are synchronous in this design
177 nCsB <= SpiRst; --these concepts are synchronous in this design
178 nCsC <= SpiRst; --these concepts are synchronous in this design
179 nCsD <= SpiRst; --these concepts are synchronous in this design
180
181 TransferComplete <= SpiRst; --these concepts are synchronous in this design
182
183 --Read A/D:
184 process (clk, rst, WriteDac, SpiXferComplete)
185 begin
186
187 if (rst = '1') then
188
189 SpiRst <= '1';
190 LastWriteDac <= '0';
191 LastSpiXferComplete <= '0';
192 DacWriteOutA_i <= x"000000";
193 DacWriteOutB_i <= x"000000";
194 DacWriteOutC_i <= x"000000";
195 DacWriteOutD_i <= x"000000";
196
197 else
198
199 if ( (clk'event) and (clk = '1') ) then
200
201 --Follow Drdy
202 if (WriteDac /= LastWriteDac) then
203
204 LastWriteDac <= WriteDac;
205
206 --Here we go...
207 if (WriteDac = '1') then
208
209 --Latch inputs & initiate reading the data.
210 DacWriteOutA_i <= DacWriteOutA;
211 DacWriteOutB_i <= DacWriteOutB;
212 DacWriteOutC_i <= DacWriteOutC;
213 DacWriteOutD_i <= DacWriteOutD;
214 SpiRst <= '0';
215
216 end if;
217
218 else
219
220 --Wait for Spi xfer to complete, then grab the sample and we're done
221 if (SpiXferComplete /= LastSpiXferComplete) then
222
223 LastSpiXferComplete <= SpiXferComplete;
224
225 if (SpiXferComplete = '1') then
226
227 --Grab read back
228 DacReadbackA <= DacReadbackA_i;
229 DacReadbackB <= DacReadbackB_i;
230 DacReadbackC <= DacReadbackC_i;
231 DacReadbackD <= DacReadbackD_i;
232
233 --turn off spi master bus
234 SpiRst <= '1';
235
236 end if;
237
238 end if;
239
240 end if;
241
242 end if;
243
244 end if;
245
246 end process;
247
248end SpiDacQuad;
