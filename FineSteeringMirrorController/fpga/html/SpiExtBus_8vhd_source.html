79 component SpiMasterPorts is
80 generic (
81 CLOCK_DIVIDER : integer := 4;
82 BYTE_WIDTH : natural := 1;
83 CPOL : std_logic := '0'--;
84 );
85 port
86 (
87 clk : in std_logic;
88 rst : in std_logic;
89 Mosi : out std_logic;
90 Sck : out std_logic;
91 Miso : in std_logic;
92 DataToMosi : in std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
93 DataFromMiso : out std_logic_vector((BYTE_WIDTH * 8) - 1 downto 0);
94 XferComplete : out std_logic--;
95 );
96 end component;
97
98 signal SpiRst : std_logic; --kicks off / inhibits transfer of a sample out of the A/D
99 signal SpiXferComplete : std_logic;
100 signal LastSpiXferComplete : std_logic;
101
102 signal SpiExtBusClk : std_logic;
103 signal LastWriteSpiExtBus : std_logic;
104 signal SpiExtBusReadback_i : std_logic_vector(7 downto 0);
105
106begin
107
108 Spi : SpiMasterPorts
109 generic map (
110 --~ CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 2000000, --2MHz works on 339...
111 CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 500000 ,
112 --~ CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 65535,
113 --~ CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 8191,
114 --~ CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 2047,
115 BYTE_WIDTH => 1,
116 --~ CPOL => '1'--, --'inverted' SCK polarity?
117 CPOL => '0'--, --'inverted' SCK polarity?
118 )
119 port map
120 (
121 clk => clk, --runs off the same clock as the A/D to keep everything nicely aligned & quiet
122 rst => SpiRst, --every sample requires a set/rst sequence to run spimaster
123 Mosi => Mosi, --we don't actually send anything to the A/D, all it needs is the sample trigger/clock
124 Sck => Sck,
125 Miso => Miso,
126 DataToMosi => SpiExtBusWriteOut, --we don't actually send anything to the A/D
127 DataFromMiso => SpiExtBusReadback_i,
128 XferComplete => SpiXferComplete--,
129 );
130
131 nCs <= SpiRst; --these concepts are synchronous in this design
132
133 --~ TransferComplete <= SpiRst; --these concepts are synchronous in this design
134
135 --~ SpiExtBusReadReady <= SpiXferComplete;
136
137 --Read A/D:
138 process (clk, rst, WriteSpiExtBus, SpiXferComplete)
139 begin
140
141 if (rst = '1') then
142
143 SpiRst <= '1';
144 LastWriteSpiExtBus <= '0';
145 LastSpiXferComplete <= '0';
146
147 else
148
149 if ( (clk'event) and (clk = '1') ) then
150
151 --Follow Drdy
152 if (WriteSpiExtBus /= LastWriteSpiExtBus) then
153
154 LastWriteSpiExtBus <= WriteSpiExtBus;
155
156 --Here we go...
157 if (WriteSpiExtBus = '1') then
158
159 SpiExtBusReadReady <= '0';
160
161 --Initiate reading the data.
162 SpiRst <= '0';
163
164 end if;
165
166 else
167
168 --Wait for Spi xfer to complete, then grab the sample and we're done
169 if (SpiXferComplete /= LastSpiXferComplete) then
170
171 LastSpiXferComplete <= SpiXferComplete;
172
173 if (SpiXferComplete = '1') then
174
175 --Grab read back
176 SpiExtBusReadback <= SpiExtBusReadback_i;
177
178 SpiExtBusReadReady <= '1';
179
180 --~ --turn off spi master bus
181 SpiRst <= '1';
182
183 end if;
184
185 end if;
186
187 --~ --Wait for Spi xfer to complete, then shut it off.
188 --~ if ( (SpiXferComplete = '1') and (LastSpiXferComplete = '1') ) then
189
190 --~ --turn off spi master bus
191 --~ SpiRst <= '1';
192
193 --~ end if;
194
195 end if;
196
197 end if;
198
199 end if;
200
201 end process;
202
203end SpiExtBus;
