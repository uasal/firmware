94 --~ signal CurrentStep_i : std_logic_vector(15 downto 0);
95 signal CurrentStep_i : signed(15 downto 0);
96 signal Direction : std_logic;
97 signal MotorStopped : std_logic;
98 signal MotorStep_i : std_logic;
99 signal InTransit : std_logic;
100

106 --Constants of the right length...
107 --~ constant DeltaSaturated : std_logic_vector(MAX_CLOCK_BITS_DELTA - 1 downto 0) := std_logic_vector(to_unsigned((2 ** (MAX_CLOCK_BITS_DELTA - 1)) - 1, MAX_CLOCK_BITS_DELTA));
108
109begin
110
111 StepOneShot : OneShotPorts
112 generic map (
113 CLOCK_FREQHZ => CLOCK_FREQHZ,
114 DELAY_SECONDS => MOTOR_STEP_SECONDS,
115 SHOT_RST_STATE => '0',
116 SHOT_PRETRIGGER_STATE => '0'
117 )
118 port map (
119 clk => clk,
120 rst => MotorStopped,
121 shot => MotorStep_i--,
122 );
123
124 StepperMotor : FourWireStepperMotorPorts
125 port map
126 (
127 clk => clk,
128 rst => rst,
129 Direction => Direction,
130 Step => MotorStep_i,
131 MotorAPlus => MotorAPlus_i,
132 MotorAMinus => MotorAMinus_i,
133 MotorBPlus => MotorBPlus_i,
134 MotorBMinus => MotorBMinus_i--,
135 );
136
137 CurrentStep <= CurrentStep_i;
138 MotorStepEdge <= MotorStep_i;
139
140 --We really don't need static holding force in most use-cases, it wastes power and heats things up!
141 MotorAPlus <= MotorAPlus_i when (InTransit = '1') else '0';
142 MotorAMinus <= MotorAMinus_i when (InTransit = '1') else '0';
143 MotorBPlus <= MotorBPlus_i when (InTransit = '1') else '0';
144 MotorBMinus <= MotorBMinus_i when (InTransit = '1') else '0';
145
146 -- Master clock drives most logic
147 process (clk, rst)
148 begin
149 if (rst = '1') then
150
151 CurrentStep_i <= x"0000";
152 Direction <= '0';
153 MotorStopped <= '1';
154 InTransit <= '0';
155
156 else
157
158 if ( (clk'event) and (clk = '1') ) then
159
160 if (CurrentStep_i /= SeekStep) then
161
162 InTransit <= '1';
163
164 if (SeekStep > CurrentStep_i) then Direction <= '1'; else Direction <= '0'; end if;
165
166 if (MotorStep_i = '0') then
167
168 MotorStopped <= '0';
169
170 else
171
172 if (SeekStep > CurrentStep_i) then CurrentStep_i <= CurrentStep_i + x"0001"; else CurrentStep_i <= CurrentStep_i - x"0001"; end if;
173
174 MotorStopped <= '1';
175
176 end if;
177
178 else
179
180 InTransit <= '0';
181
182 MotorStopped <= '1';
183
184 end if;
185
186 end if;
187
188 end if;
189
190 end process; --(clock)
191
192end FourWireStepperMotorDriver;
