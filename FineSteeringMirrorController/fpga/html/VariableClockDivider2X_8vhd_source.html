47 signal ClkDiv : natural range 0 to ((2**WIDTH_BITS) - 1) := 0;
48 signal clko_i : std_logic;

50begin

54 process (clki, rst, rst_count)
55 begin
56
57 if (rst = '1') then
58
59 ClkDiv <= to_integer(unsigned(rst_count));

61
62 else
63
64 if ( (clki'event) and (clki = '1') ) then
65
66 --Generate the master DE-sampling clock

68
69 ClkDiv <= ClkDiv + 1;
70
71 else
72
73 ClkDiv <= 0;
74
75 end if;
76

78
79 clko_i <= DIVOUT_RST_STATE;
80
81 else
82
83 clko_i <= not(DIVOUT_RST_STATE);
84
85 end if;

87 end if;

89 end if;
