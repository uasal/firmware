108 --Run the SpiExtBus addr output cycle:
109 process (clk, rst, SpiExtBusAddrTxCurrentState)
110 begin
111
112 if (rst = '1') then
113
114 LatchTxdSpiExtBusAddr <= '0';
115 StartTxdSpiExtBusAddr <= '0';
116 SendingSpiExtBusAddr <= '0';
117 SpiExtBusAddrTxNextState <= Idle;
118
119 else
120
121 if ( (clk'event) and (clk = '1') ) then
122
123 SpiExtBusAddrTxCurrentState <= SpiExtBusAddrTxNextState;
124
125 case SpiExtBusAddrTxCurrentState is
126
127 when Idle =>
128
129 if (SendSpiExtBusAddr = '1') then
130
131 LatchTxdSpiExtBusAddr <= '1';
132
133 end if;
134
135 if ( (SendSpiExtBusAddr = '0') and (LatchTxdSpiExtBusAddr = '1') ) then
136
137 SpiExtBusAddrTxNextState <= Load;
138
139 end if;
140
141 when Load =>
142
143 StartTxdSpiExtBusAddr <= '1';
144
145 LatchTxdSpiExtBusAddr <= '0';
146
147 SendingSpiExtBusAddr <= '1';
148
149 SpiExtBusAddrTxNextState <= Loaded;
150
151 when Loaded =>
152
153 if (SpiExtBusAddrTxdInProgress = '1') then
154
155 StartTxdSpiExtBusAddr <= '0';
156
157 SpiExtBusAddrTxNextState <= Transmit;
158
159 end if;
160
161 when Transmit =>
162
163 if (SpiExtBusAddrTxdInProgress = '0') then
164
165 SendingSpiExtBusAddr <= '0';
166
167 SpiExtBusAddrTxNextState <= Idle;
168
169 end if;
170
171 when others => -- ought never to get here...
172
173 LatchTxdSpiExtBusAddr <= '0';
174 StartTxdSpiExtBusAddr <= '0';
175 SendingSpiExtBusAddr <= '0';
176 SpiExtBusAddrTxNextState <= Idle;
177
178 end case;
179
180 end if;
181
182 end if;
183
184 end process;
185
186end SpiExtBusAddrTx;
