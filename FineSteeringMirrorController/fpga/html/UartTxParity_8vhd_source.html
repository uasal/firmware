47 signal BitCnt : unsigned(3 downto 0); -- bit counter
48 signal LastGo : std_logic := '0';
49 signal ParityBit : std_logic := '0';
50
51begin
52
53 process(Clk, Reset, Go)
54 constant CntOne: Unsigned(3 downto 0):="0001";
55 begin
56 if (Reset = '1') then
57 TxD <= '1';
58 BitCnt <= "1110";
59 Busy <= '0';
60 ParityBit <= '0';
61 LastGo <= '0';
62 else
63 if (Rising_Edge(Clk)) then
64
65 if (Go /= LastGo) then
66 LastGo <= Go;
67 if (Go = '1') then BitCnt <= "1111"; end if;
68 end if;
69
70 case BitCnt is
71 when "1111" => --start bit
72 TxD <= '0';
73 BitCnt <= "0000";
74 Busy <= '1';
75 ParityBit <= '0';
76 when "0000" | "0001" | "0010" | "0011" |
77 "0100" | "0101" | "0110" |
78 "0111" => --data bits, lsb first
79 TxD <= Data(to_integer(BitCnt(2 downto 0)));
80 ParityBit <= ParityBit xor Data(to_integer(BitCnt(2 downto 0)));
81 BitCnt <= BitCnt + CntOne;
82 when "1000" => --parity bit
83 if (PARITY_EVEN = '1') then
84 TxD <= ParityBit;
85 else
86 TxD <= not(ParityBit);
87 end if;
88 BitCnt <= BitCnt + CntOne;
89 when "1001" => --stop bit
90 TxD <= '1';
91 BitCnt <= BitCnt + CntOne;
92 when others => --idle (i.e. "1010+")
93 TxD <= '1';
94 Busy <= '0';
95 end case;
96 end if;
97 end if;
98 end process;
99
100end Behaviour;
