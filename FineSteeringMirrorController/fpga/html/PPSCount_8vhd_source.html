49architecture PPSCount of PPSCountPorts is
50
51 signal LastPPS : std_logic;
52 signal InvalidatePPSCount : std_logic := '0';
53 signal PPSAccum_i : std_logic_vector(31 downto 0);
54 signal PPSAccumCycles : natural range 0 to PPS_ACCUM_CYCLES := 0;
55
56begin
57
58 PPSCounter <= PPSAccum_i;
59
60 -- Master clock drives most logic
61 process (PPSReset, clk, PPS)
62 begin
63
64 --async reset
65 if (PPSReset = '1') then
66
67 --Reset cycle count
68 PPSAccumCycles <= 0;
69
70 PPSDetected <= '0';
71
72 --Reset latch
73 PPSAccum <= x"00000000";
74
75 --Invalidate next accum cycle, so we don't get a good looking count that's missing some small part of the second and throws off filters...
76 InvalidatePPSCount <= '1';
77
78 else
79
80 if ( (clk'event) and (clk = '1') ) then
81
82 --rising edge of pps - run pps cycle counter
83 if (PPS /= LastPPS) then
84
85 LastPPS <= PPS;
86
87 PPSDetected <= '1';
88
89 if (PPS = '1') then
90
91 --Move on to next PPS cycle
92 --~ PPSAccumCycles <= PPSAccumCycles + 1;
93
94 --have we reached the end of our accumulation?
95 --~ if (PPSAccumCycles >= PPS_ACCUM_CYCLES - 1) then
96
97 --Reset cycle count
98 --~ PPSAccumCycles <= 0;
99
100 --Reset accumulator
101 PPSAccum_i <= x"00000000";
102
103 if (InvalidatePPSCount = '1') then
104
105 InvalidatePPSCount <= '0';
106
107 PPSAccum <= x"00000000";
108
109 else
110
111 PPSAccum <= PPSAccum_i + x"00000002"; --latch output - since each edge of PPS takes 1 clock, the actual count is two more than the accumulator has in it
112
113 end if;
114
115 --~ end if;
116
117 end if;
118
119 --keep accumulating
120 else
121
122 PPSAccum_i <= PPSAccum_i + x"00000001";
123
124 end if;
125
126 end if;
127
128 end if;
129
130 end process; --(clock)
131
132end PPSCount;
