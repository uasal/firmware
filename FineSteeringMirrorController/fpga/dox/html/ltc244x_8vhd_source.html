98  --nDrdy is inverted (obviously)
99  constant DataReady : std_logic := '0';
100  constant NoDataReady : std_logic := '1';
101  signal LastnDrdy : std_logic; --to grab edge
102  signal LastReadChannel : std_logic; --to grab edge
103  signal DataFromMiso : std_logic_vector(31 downto 0);

109 begin
110 
111  Spi : SpiMasterPorts
112  generic map (
113  CLOCK_DIVIDER => MASTER_CLOCK_FREQHZ / 5000000 , --5MHz seems reasonable for a 20MHz part
114  BYTE_WIDTH => 4,
115  CPOL => '0'--, --'normal' SCK polarity
116  )
117  port map
118  (
119  clk => clk, --runs off the same clock as the A/D to keep everything nicely aligned & quiet
120  rst => SpiRst, --every sample requires a set/rst sequence to run spimaster
121  Mosi => Mosi,
122  Sck => Sck,
123  Miso => Miso,
124  DataToMosi => DataToMosi,
125  DataFromMiso => DataFromMiso,
126  XferComplete => SpiXferComplete--,
127  );
128 
129  --~ nCs <= SpiRst; --these concepts are synchronous in this design
130 
131  --Read A/D:
132  process (clk, rst, nDrdy, SpiXferComplete)
133  begin
134 
135  if (rst = '1') then
136 
137  SpiRst <= '1';
138 
139  else
140 
141  if ( (clk'event) and (clk = '1') ) then
142 
143  --ReadChannel kicks the whole thing off signal controls Spi master transfers
144  if (ReadChannel /= LastReadChannel) then
145 
146  LastReadChannel <= ReadChannel;
147 
148  if (ReadChannel = '1') then
149 
150  nCs <= '0'; --fire up the spi bus to get the sample off the a/d
151 
152  SampleLatched <= '0'; --no sample ready now.
153 
154  end if;
155 
156  end if;
157 
158  --Dataready signal controls Spi master transfers
159  if (nDrdy /= LastnDrdy) then
160 
161  LastnDrdy <= nDrdy;
162 
163  if (nDrdy = DataReady) then
164 
165  SpiRst <= '0'; --fire up the spi bus to get the sample off the a/d
166 
167  else
168 
169  SpiRst <= '1'; --new sample incoming, reset bus
170 
171  end if;
172 
173  end if;
174 
175  --When spi xfer is complete, grab the data
176  if (SpiXferComplete /= LastSpiXferComplete) then
177 
178  LastSpiXferComplete <= SpiXferComplete;
179 
180  if (SpiXferComplete = '1') then
181 
182  Sample <= DataFromMiso;
183 
184  nCs <= '1'; --fire up the spi bus to get the sample off the a/d
185 
186  SampleLatched <= '1'; --no sample ready now.
187 
188  end if;
189 
190  end if;
191 
192  end if;
193 
194  end if;
195 
196  end process;

198 end ltc244x_i;
