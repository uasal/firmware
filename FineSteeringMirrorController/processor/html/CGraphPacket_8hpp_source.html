367 {

373 {
374 Mappings[i].ControllerBoardIndex = ControllerBoardIndex;
375 Mappings[i].DacIndex = DacIndex;
376 Mappings[i].DacChannel = DacChannel;
377
378 DacChannel++;
379 if (DacChannel >= DMActuatorsPerDac)
380 {
381 DacChannel = 0;
382 DacIndex++;
383 if (DacIndex >= DMMDacsPerControllerBoard)
384 {
385 DacIndex = 0;
386 ControllerBoardIndex++;
387 if (ControllerBoardIndex >= DMMaxControllerBoards)
388 {
389 //We really really really shouldn't get here, but just in case we do it's better than crashing...
390 ControllerBoardIndex = 0;
