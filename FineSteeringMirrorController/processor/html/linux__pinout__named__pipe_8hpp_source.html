58 virtual int init(const uint32_t ReadWriteMask, const char* named_pipe_path_name)
59 {
60 deinit(); //close named_pipe if open
61
62 //~ printf("\nlinux_pinout_named_pipe::init(): silent: %c.\n", silent?'Y':'N');
63
64 //~ if (!silent) { printf("\nlinux_pinout_named_pipe::init() : opening named_pipe \"%s\".\n", named_pipe_path_name); }
65
66 //We need fifo to be readable by nobody a lot...
67 umask(S_IRUSR | S_IWUSR | S_IXUSR | S_IRGRP | S_IWGRP | S_IXGRP | S_IROTH | S_IWOTH | S_IXOTH);
68
69 if (0 != mkfifo(named_pipe_path_name, S_IRUSR | S_IWUSR | S_IXUSR | S_IRGRP | S_IWGRP | S_IXGRP | S_IROTH | S_IWOTH | S_IXOTH))
70 {
71 if (EEXIST != errno) //If it's there already from our last run, then fine.
72 {
73 //~ printf("\nlinux_pinout_named_pipe::init() : error in mkfifo(): %ld.\n", (long int)errno);
74 if (!silent) { perror("\nlinux_pinout_named_pipe::init() : error in mkfifo()"); }
75 return(errno);
76 }
77 }
78
79 fd = open(named_pipe_path_name, (int)ReadWriteMask | O_NONBLOCK);

107 virtual bool dataready() const
108 {
109 fd_set pipeset;
110 struct timeval nowait;
111 memset((char *)&nowait,0,sizeof(nowait));
112 if (-1 == fd) { return(false); } //open?
113 FD_ZERO(&pipeset);
114 FD_SET(fd, &pipeset);
115 int result = select(fd + 1, &pipeset, NULL, NULL, &nowait);
116 if (result < 0)
117 {
118 //~ if (!silent) { printf("\nlinux_pinout_named_pipe::dataready(): select() failed with %ld.\n", (long int)errno); }
119 if (!silent) { perror("\nlinux_pinout_named_pipe::dataready(): select() failed"); }
120 }
121 else if (result == 1)
122 {
123 if (FD_ISSET(fd, &pipeset)) // The pipeet has data. For good measure, it's not a bad idea to test further
124 {
125 return(true);
126 }
127 }
128 return(false);
129 }
