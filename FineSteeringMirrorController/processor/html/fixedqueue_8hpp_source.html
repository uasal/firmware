31template<typename T, size_t queue_size> class fixedqueue
32{
33 private:

35 T data[queue_size] __attribute__((aligned(4)));

37 size_t volatile begin;

40 size_t next(const size_t in) const
41 {
42 size_t tNxt ;
43// joeb
44// tNxt = (in == (queue_size - 1)) ? 0 : in + 1 ;
45 tNxt = ( in + 1 ) % queue_size ;
46 return tNxt ;
47 }
48
49 public:
50
51 fixedqueue(): begin(0), end(0) { }
52
53 bool empty() const { return begin == end; }
54
55 bool full() const { return (next(end) == begin); }
56
57// joeb
58// size_t max() const { return(queue_size); }
59 size_t max() const { return(queue_size -1); }
60
61 size_t len() const
62 {
63// joeb
64// size_t temp(end-begin);
65// if (temp < 0) temp += queue_size;
66 size_t temp = ( queue_size + end - begin ) % queue_size ;
67 return temp;
68 }
69
70 void clear() { begin = 0; end = 0; }
71 T const& front() const { return data[begin]; }
72 T const& back() const { return data[end ? (end-1) : (queue_size-1)]; }
73
74 bool pop(T& element)
75 {
76 if ( empty() ) { return(false); }
77 element = data[begin];
78 begin = next(begin);
79 return(true);
80 }
81
82 bool push(T const& element)
83 {
84 if ( full() ) { return(false); }
85 data[end] = element;
86 end = next(end);
87 return(true);
88 }
