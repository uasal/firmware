96int8_t BinaryFSMAdcsFloatingPointCommand(const uint32_t Name, char const* Params, const size_t ParamsLen, const void* Argument)
97{
98 printf("\nBinaryFSMAdcsCommand processing(%u)...\n\n", ParamsLen);
99
100 double AdcVals[3];
101 AdcAccumulator A, B, C;
102 A = FSM->AdcAAccumulator;
103 B = FSM->AdcBAccumulator;
104 C = FSM->AdcCAccumulator;
105 AdcVals[0] = (8.192 * ((A.Samples - 0) / A.NumAccums)) / 16777216.0;
106 AdcVals[1] = (8.192 * ((B.Samples - 0) / B.NumAccums)) / 16777216.0;
107 AdcVals[2] = (8.192 * ((C.Samples - 0) / C.NumAccums)) / 16777216.0;
108 printf("\nBinaryFSMAdcsFloatingPointCommand Replying (%lf, %lf, %lf)...\n\n", AdcVals[0], AdcVals[1], AdcVals[2]);
109 TxBinaryPacket(Argument, CGraphPayloadTypeFSMAdcsFloatingPoint, 0, AdcVals, 3 * sizeof(double));
110
111 return(ParamsLen);

114int8_t BinaryFSMDacsFloatingPointCommand(const uint32_t Name, char const* Params, const size_t ParamsLen, const void* Argument)
115{
116 double VA = 0.0, VB = 0.0, VC = 0.0;
117 unsigned long A = 0, B = 0, C = 0;
118
119 if (ParamsLen >= (3 * sizeof(double)))
120 {
121 const double* DacSetpoints = (const double*)Params;
122
123 VA = DacSetpoints[0];
124 VB = DacSetpoints[1];
125 VC = DacSetpoints[2];
126
127 A = (VA * (double)(0x00FFFFFFUL) * 60.0) / 4.096;
128 B = (VB * (double)(0x00FFFFFFUL) * 60.0) / 4.096;
129 C = (VC * (double)(0x00FFFFFFUL) * 60.0) / 4.096;
130
131 printf("\n\nBinaryFSMDacsCommand: Setting to: %3.1lf (%lx), %3.1lf (%lx), %3.1lf (%lx).\n", VA, A, VB, B, VC, C);
132
133 FSM->DacASetpoint = DacSetpoints[0];
134 FSM->DacBSetpoint = DacSetpoints[1];
135 FSM->DacCSetpoint = DacSetpoints[2];
136 }
137
138 uint32_t DacSetpoints[3];
139
140 A = FSM->DacASetpoint;
141 B = FSM->DacBSetpoint;
142 C = FSM->DacCSetpoint;
143
144 VA = 4.096 * (double)(A) / ((double)(0x00FFFFFFUL) * 60.0);
145 VB = 4.096 * (double)(B) / ((double)(0x00FFFFFFUL) * 60.0);
146 VC = 4.096 * (double)(C) / ((double)(0x00FFFFFFUL) * 60.0);
147
148 DacSetpoints[0] = VA;
149 DacSetpoints[1] = VB;
150 DacSetpoints[2] = VC;
151
152 printf("\n\nBinaryFSMDacsCommand: Replying: %3.1lf (%lx), %3.1lf (%lx), %3.1lf (%lx).\n", VA, A, VB, B, VC, C);
153 TxBinaryPacket(Argument, CGraphPayloadTypeFSMDacsFloatingPoint, 0, DacSetpoints, 3 * sizeof(double));
154
155 return(ParamsLen);
156}
