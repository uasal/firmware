198 void Init()
199 {
200 for (uint8_t i = 0; i < 10; i++)
201 {
202 uint8_t AdcInit = Adc.Init(GpioDirections);
203 if (Adc.InitOK == AdcInit)
204 {
205 AdcFound = true;
206 break;
207 }
208 else
209 {
210 formatf("\nMonitorAdc: Error initializing Adc: %u [0x%.2X].\n", AdcInit, AdcInit);
211 }
212 }
213 if (AdcFound)
214 {
215 Adc.ClearScanChannels();
216 Adc.AddScanChannel(P1V2Channel);
217 Adc.AddScanChannel(P2V2Channel);
218 Adc.AddScanChannel(P28VChannel);
219 Adc.AddScanChannel(P2V5Channel);
220 Adc.AddScanChannel(P6VChannel);
221 Adc.AddScanChannel(P5VChannel);
222 Adc.AddScanChannel(P3V3DChannel);
223 Adc.AddScanChannel(P4V3Channel);
224 Adc.AddScanChannel(P2I2Channel);
225 Adc.AddScanChannel(P4I3Channel);
226 Adc.AddScanChannel(P6IChannel);
227 Adc.AddScanChannel(Aux0Channel);
228 Adc.AddScanChannel(Aux1Channel);
229 Adc.AddScanChannel(Aux2Channel);
230 Adc.AddScanChannel(AmbientLightChannel);
231 Adc.AddScanChannel(TemperatureChannel);
232 Adc.CommitScanChannels();
233 Adc.StartChannelScan();
234 //Take start pin high to initiate auto-scan
235 {
236 Adc.WriteRegister(ads1258details::register_gpiod, StartPinMask);
237 //~ if (Monitor)
238 {
239 formatf("\nMonitorAdc: Starting A/D auto-scan; wrote 0x%.2X to gpio's, readback: 0x%.2X\n", StartPinMask, Adc.ReadRegister(ads1258details::register_gpiod));
240 fflush(stdout);
241 }
242 }
243 }

250 void Process()
251 {
252 if (AdcFound)
253 {
254 //~ we don't have anything connected to gpios but might in the future, and dm does if this code gets cut&pasted: GpioInputs = Adc.ReadRegister(ads1258details::register_gpiod);
255
256 ads1258details::ads1258sample sample;
257
258 for(size_t CurrentChan = 0; CurrentChan < ads1258details::ads1258numchannels; CurrentChan++)
259 {
260 ProcessAllUarts();
261
262 Adc.AutoScan();
263
264 if (Adc.IsScanChannel(CurrentChan))
265 {
266 ProcessAllUarts();
267
268 Adc.GetLastSample(CurrentChan, sample);
269
270 ProcessAllUarts();
271
272 if ( (sample.status.isbrownout) || (sample.status.isclipped) )
273 {
274 if (Monitor) { ::formatf("\nMonitorAdc: ch %u bad status: 0x%.2X\n", sample.status.channel, sample.status.all); }
275 }
276 else
277 {
278 if (sample.status.isnew)
279 {
280 if (Monitor) { ::formatf("\nMonitorAdc: ch %u reads: %lf Volts [%lu lsb's]\n", sample.status.channel, Adc.CountsToVolts(sample.sample), sample.sample); }
281
282 switch(sample.status.channel)
283 {
284 case P1V2Channel : { P1V2 = (sample.sample); break; }
285 case P2V2Channel : { P2V2 = (sample.sample); break; }
286 case P28VChannel : { P28V = (sample.sample); break; }
287 case P2V5Channel : { P2V5 = (sample.sample); break; }
288 case P6VChannel : { P6V = (sample.sample); break; }
289 case P5VChannel : { P5V = (sample.sample); break; }
290 case P3V3DChannel : { P3V3D = (sample.sample); break; }
291 case P4V3Channel : { P4V3 = (sample.sample); break; }
292 case P2I2Channel : { P2I2 = (sample.sample); break; }
293 case P4I3Channel : { P4I3 = (sample.sample); break; }
294 case P6IChannel : { P6I = (sample.sample); break; }
295 case Aux0Channel : { Aux0 = (sample.sample); break; }
296 case Aux1Channel : { Aux1 = (sample.sample); break; }
297 case Aux2Channel : { Aux2 = (sample.sample); break; }
298 case AmbientLightChannel : { AmbientLight = (sample.sample); break; }
299 case TemperatureChannel : { Temperature = (sample.sample); break; }
300
301 default: { break; } //do nothing; we can scan all 16 channels and discard the unused ones.
302 }
303 }
304 else
305 {
306 if (Monitor) { ::formatf("\nMonitorAdc: ch %u no new sample this scan.\n", sample.status.channel); }
307 }
308 }
309 }
310 }
311 }
