197 bool Process()// override
198 {
199 bool Processed = false;
200
201 //Of note: the DataLen should only grow as we process, since there is no other consumer removing bytes from the buffer, and we're not going to remove any until it overflows or we find the end of a packet.
202
203 //New char?
204 size_t DataLen = Data.Depth();
205 if (LastDataLen == DataLen)
206 {
207 return(false);
208 }
209 int32_t NewDataLen = DataLen - LastDataLen;
210 if (NewDataLen < 0)
211 {
212 NewDataLen = 0;
213 }
214 LastDataLen = DataLen;
215 if (0 == LastDataLen)
216 {
217 return(false);
218 }
219
220 //Packet End?? We have nothing to do until we have the end of a packet!
221 //Need at least a Footer's worth of data...
222 if (LastDataLen >= Packet.FooterLen())
223 {
224 //Where do we start searching for the Footer?
225 int32_t FooterSearchStartPos = LastDataLen - NewDataLen;
226 FooterSearchStartPos -= Packet.FooterLen(); //If we just got the final byte(s) of the footer, we need to look behind in the buffer from the start of the new data to actually find the footer!
227 if (FooterSearchStartPos < 0)
228 {
229 FooterSearchStartPos = 0; //But that could underrun the buffer so we truncate...
230 }
231
232 //Did we get it??
233 PacketEndPos = LastDataLen;
234 if (Packet.FindPacketEndPos(Data, FooterSearchStartPos, PacketEndPos))
235 {
236 //Oh, things just got interesting!
237
238 // if (debug) { ::formatf("\n\nBinaryUart: Packet end detected! Searching for start...\n\r"); }
239 //~ InPacket = true;
240
241 if (Packet.FindPacketStartPos(Data, 0, PacketStartPos))
242 {
243 //Oh, now things just got really really interesting!!
244 PayloadLen = Packet.PayloadLen(Data, PacketStartPos);
245 HeaderLen = Packet.HeaderLen();
246 FooterLen = Packet.FooterLen();
247
248 // Validate packet
249 if (Packet.IsValid(Data, PacketStartPos, PacketEndPos))
250 {
251 // Confirm that the serial number matches or is a broadcast
252 if ( (SerialNum == InvalidSerialNumber) || (Packet.IsBroadcastSerialNum(Data, PacketStartPos, PacketEndPos)) || (SerialNum == Packet.SerialNum(Data, PacketStartPos, PacketEndPos)) )
253 {
254 // Just look at each command, and exectute it if the input line matches.
255 bool CmdFound = false;
256
257 // Search for a matching command in the packet
258 // Inefficient to go through whole list of commands
259 // Idea: last nibble or byte (depending on number of commands) corresponds to position of command
260 // and can be used as index to Cmds[i].Response(...)
261 // This only happens once for each full packet, so might not be much time savings
262 for (size_t i = 0; i < NumCmds; i++)
263 {
264 // Check if command in buffer matches the defined command name
265 if (Packet.DoesPayloadTypeMatch(Data, PacketStartPos, PacketEndPos, Cmds[i].Name))
266 {
267 //strip the part of the line with the arguments to this command (chars following command) for compatibility with the parsing code, the "params" officially start with the s/n
268
269 //**warning, if packet wraps the end of the ring buffer, and the cmd handler dereferences it, this is gonna overrun the buffer and crash**
270 //(but we're supposed to just be passing the address of the data in the buffer back to the fpga to copy it to the d/a's, not playing with the data)
271 int warning_fix_copy_to_safe_buffer = 37;
272 const char* Params = (char*)(&(Data[PacketStartPos + HeaderLen]));
273
274 // Execute the command's response function
275 // We already figured out the payload length, so we don't need to call Packet.PayloadLen
276 // to get it again. Comment out and use simpler Cmds[i].Respons(...) below
277 Cmds[i].Response(Cmds[i].Name, Params, Packet.PayloadLen(Data, PacketStartPos), (void*)this);
278 CmdFound = true;
279 Processed = true;
280 }
281 }
282
283 // If no command was found, trigger the unhandled packet callback
284 if (!CmdFound)
285 {
286 //~ if (debug) { ::formatf("\n\nBinaryUart: Unmatched command 0x%.8lX!\n", PacketHeader->PayloadTypeToken); }
287 if (debug)
288 {
289 ::formatf("\n\nBinaryUart: Unmatched command 0x%.8lX! NumCmds: %lu\n", Packet.PayloadType(Data, PacketStartPos, PacketEndPos), (unsigned long)NumCmds);
290 }
291
292 Callbacks.UnHandledPacket(reinterpret_cast<IPacket*>(&(const uint8_t*)(Data.Data)[PacketStartPos]), PacketEndPos - PacketStartPos);
293 }
294 else
295 {
296 if (debug)
297 {
298 //~ ::formatf("\n\nBinaryUart: Got Packet! \n");
299 //~ PacketHeader->formatf();
300 //~ ::formatf(" <");
301 //~ for (size_t k = PacketStartPos; k < PacketEndPos; k++)
302 //~ {
303 //~ ::formatf(":%0.2X", Data.Data[k]);
304 //~ }
305 //~ ::formatf(">\n\n\n");
306 }
307 }
308 }
309 else
310 // If serial number does not match, packet is unhandled
311 {
312 if (debug)
313 {
314 ::formatf("\n\nBinaryUart: Packet received, but SerialNumber comparison failed (expected: 0x%.8lX; got: 0x%.8lX).\n\r", SerialNum, Packet.SerialNum(Data.Data, PacketStartPos, PacketEndPos));
315 }
316
317 Callbacks.UnHandledPacket(reinterpret_cast<IPacket*>(&(const uint8_t*)(Data.Data)[PacketStartPos]), PacketEndPos - PacketStartPos);
318 }
319
320 // Notify that every packet is processed, even if unmatched
321 Callbacks.EveryPacket(reinterpret_cast<IPacket*>(&(const uint8_t*)(Data.Data)[PacketStartPos]), PacketEndPos - PacketStartPos);
322
323 //Ok, now we have to figure out how to remove the packet from the buffer...
324 Data.Pop(PacketEndPos);
325 //~ InitFast(SerialNum);
326 }
327 else
328 // If packet is invalid, trigger the invalid packet callback
329 {
330 if (debug)
331 {
332 ::formatf("\n\nBinaryUart: Packet received, but invalid.\n\r");
333 }
334
335 Callbacks.InvalidPacket((uint8_t*)(Data.Data), LastDataLen);
336 }
337
338 }
339 }
340 }

342 return(true); //We just want to know if there's chars in the buffer to put threads to sleep or not...

356 virtual void TxBinaryPacket(const uint16_t PayloadType, const uint32_t SerialNumber, const void* PayloadData, const size_t PayloadLen) const
357 {
358 uint8_t TxBuffer[TxBufferLenBytes];
359 size_t PktLen = Packet.MakePacket(TxBuffer, TxBufferLenBytes, PayloadData, PayloadType, PayloadLen);
360
361 // Transmit each byte of the packet through the UART pinout
362 for (size_t i = 0; i < PktLen; i++)
363 {
364 Pinout.putcqq(TxBuffer[i]);
365 }
366
367 // Debug output: log the packet type, length, and contents in hex format
368 if (debug)
369 {
370 ::formatf("\n\nBinary Uart: Sending packet(%u, %u): ", PayloadType, PayloadLen);
371 for(size_t i = 0; i < PktLen; i++)
372 {
373 printf("%.2X:", TxBuffer[i]);
374 }
375 printf("\n\n");
376 }
