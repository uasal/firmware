391int8_t CirclesCommand(char const* Name, char const* Params, const size_t ParamsLen, const void* Argument)
392{
393 size_t cycle = 0;
394 unsigned long daca = 0;
395 unsigned long dacb = 0;
396 unsigned long dacc = 0;
397 int key = 0;
398
399 double radius = 1.0;
400 double delayinms = 1.0; //1ms
401 sscanf(Params, "%lf,%lf", &radius, &delayinms);
402 if (radius < 0.0) { radius = 0.0; }
403 if (radius > 1.0) { radius = 1.0; }
404 if (delayinms < 0.001) { delayinms = 0.001; }
405 if (delayinms > 10000) { delayinms = 10000; }
406
407 printf("\n\nCircles: RunCircle(%lf, %lfms)...\n", radius, delayinms);
408
409 while(true)
410 {
411 cycle++;
412
413 //Update the D/A's every so often
414 {
415 FSM->DacASetpoint = daca;
416 FSM->DacBSetpoint = dacb;
417 //~ FSM->DacBSetpoint = 0x00AAAAAAUL;
418 //~ FSM->DacBSetpoint = 0x006FFFFFUL;
419 //~ FSM->DacBSetpoint = 0x00CFFFFFUL;
420 FSM->DacCSetpoint = dacc;
421 //~ printf("\n\nBIST: D/A's set to: %lx, %lx, %lx.\n", daca, 0x006FFFFFUL, dacc);
422
423 unsigned long rba = FSM->DacASetpoint;
424 unsigned long rbb = FSM->DacBSetpoint;
425 unsigned long rbc = FSM->DacCSetpoint;
426 printf("\n%lu, %lu, %lu", rba, rbb, rbc);
427
428
429 //~ double ang = (double)(cycle % 360);
430 double ang = (double)(cycle % 60) * 6.0;
431 //~ double rad = (ang / 360.0) * 6.28;
432 double rada = (ang / 360.0) * 6.28;
433 double radb = ((ang + 120) / 360.0) * 6.28;
434 double radc = ((ang + 240) / 360.0) * 6.28;
435 //~ double carta = ((sin(rad) + 1.0) / 2.0) * radius;
436 //~ double cartc = ((cos(rad) + 1.0) / 2.0) * radius;
437 //~ printf("\n\nBIST: Deg:%f, Rad:%f, Sin:%f, Cos:%f.\n", ang, rad, carta, cartb);
438 double carta = ((sin(rada) + 1.0) / 2.0) * radius;
439 double cartb = ((sin(radb) + 1.0) / 2.0) * radius;
440 double cartc = ((sin(radc) + 1.0) / 2.0) * radius;
441
442 daca = (unsigned long)(carta * 0x00CFFFFFUL);
443 dacb = (unsigned long)(cartb * 0x00CFFFFFUL);
444 dacc = (unsigned long)(cartc * 0x00CFFFFFUL);
445 //~ daca = (unsigned long)(carta * 0x003FFFFFUL);
446 //~ dacc = (unsigned long)(cartb * 0x003FFFFFUL);
447
448 //~ printf("\n%lu, %lu, %lu", daca, dacb, dacc);
449 }
450
451 //Quit on any keypress
452 {
453 //~ if (0 != key)
454 {
455 fflush(stdin);
456 printf("\n\nCircles: Keypress(%d); exiting.\n", key);
457 break;
458 }
459 }
460
461 //~ DelayMs(delayinms);
462 }
463
464 return(ParamsLen);
465}
