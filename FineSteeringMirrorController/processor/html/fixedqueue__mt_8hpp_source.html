32template<typename T, size_t queue_size> class fixedqueue_mt
33{
34 private:
35
36 T data[queue_size] __attribute__((aligned(4)));
37
38 std::atomic <size_t> begin;
39 std::atomic <size_t> end;
40
41 size_t next(const size_t in) const { return( (in == (queue_size - 1)) ? 0 : in + 1 ); }
42
43 public:
44
45 fixedqueue_mt(): begin(0), end(0) { }
46
47 size_t max() const { return(queue_size); }
48
49 bool empty() const { return begin.load() == end.load(); }
50 bool wasEmpty() const { return(empty()); }
51
52 bool full() const { return (next(end.load()) == begin.load()); }
53 bool wasFull() const { return(full()); }
54
55 size_t len() const
56 {
57 size_t temp(end.load()-begin.load());
58 //~ if (temp < 0) temp += queue_size;
59 if (temp >= queue_size) temp %= queue_size;
60 return temp;
61 }
62 size_t depth() const { return(len()); }
63
64 void clear() { begin.store(0); end.store(0); }
65 void flush() { return(clear()); }
66
67 bool pop(T& element)
68 {
69 if ( empty() ) { return(false); }
70 element = data[begin.load()];
71 begin.store(next(begin.load()));
72 return(true);
73 }
74
75 bool push(T const& element)
76 {
77 if ( full() ) { return(false); }
78 data[end.load()] = element;
79 end.store(next(end.load()));
80 return(true);
81 }
82};
