139  bool ProcessByte(const char c)
140  {
141  bool Processed = false;
142  size_t PacketEnd = 0;
143 
144  //~ stdio_hook_putc(c);
145  //~ ::formatf(":0x%.2X", c);
146  //~ ::formatf("\"%c\":", c);
147 
148  //Put the current character into the buffer
149  if (RxCount < RxBufferLenBytes)
150  {
151  RxCount++;
152  RxBuffer[RxCount - 1] = c;
153  }
154  else
155  {
156  if (debug) { ::formatf("\n\nBinaryUart: Buffer(%p) overflow; this packet will not fit (%zub), flushing buffer.\n", RxBuffer, RxCount); }
157 
158  Callbacks.BufferOverflow(RxBuffer, RxCount);
159 
160  InPacket = false;
161  RxCount = 0;
162  RxBuffer[0] = 0;
163  }
164 
165  //Packet Start?
166  if ( (!InPacket) && (RxCount >= Packet.HeaderLen()) )
167  {
168  if (Packet.FindPacketStart(RxBuffer, RxCount, PacketStart)) //This is wasteful, we really only need to look at the 4 newest bytes every time...
169  {
170  if (debug) { ::formatf("\n\nBinaryUart: Packet start detected! Buffering.\n"); }
171 
172  InPacket = true;
173  }
174  }
175 
176  //Packet End?
177  if ( (InPacket) && (RxCount >= (Packet.HeaderLen() + Packet.FooterLen())) )
178  {
179  if (Packet.FindPacketEnd(RxBuffer, RxCount, PacketEnd)) //This is wasteful, we really only need to look at the 4 newest bytes every time...
180  {
181  if (debug) { ::formatf("\n\nBinaryUart: Packet end detected; Looking for matching packet handlers.\n"); }
182 
183  if (RxCount >= Packet.PayloadLen(RxBuffer, RxCount, PacketStart) + Packet.HeaderLen() + Packet.FooterLen())
184  {
185  if (Packet.IsValid(RxBuffer, RxCount, PacketStart, PacketEnd))
186  {
187  if ( (SerialNum == InvalidSerialNumber) || (Packet.IsBroadcastSerialNum(RxBuffer, PacketStart, PacketEnd)) || (SerialNum == Packet.SerialNum(RxBuffer, PacketStart, PacketEnd)) )
188  {
189  //just look at each command, and exectute it if the input line matches.
190  bool CmdFound = false;
191 
192  for (size_t i = 0; i < NumCmds; i++)
193  {
194  //~ formatf("\nParseCmd(): i=%d, Line=%s (len=%d), L[0] = 0x%.2X, Cmd=", i, LineIn, strlen(LineIn), (uint16_t)(LineIn[0]));
195  //~ formatf(Cmds[i].Name);
196  //~ formatf(" CmdLen=%d\n", Cmds[i].Strlen);
197 
198  //do we match command(i)?
199  if (Packet.DoesPayloadTypeMatch(RxBuffer, RxCount, PacketStart, PacketEnd, Cmds[i].Name))
200  {
201  //~ ::formatf("\n\nBinary Uart: PayloadTypeMatch. (%u, %u): ", PacketStart, Packet.PayloadOffset());
202  //~ for(size_t i = 0; i < PacketEnd + 4; i++) { printf("%.2X:", RxBuffer[i]); }
203  //~ printf("\n\n");
204 
205  //strip the part of the line with the arguments to this command (chars following command) for compatibility with the parsing code, the "params" officially start with the s/n
206  const char* Params = reinterpret_cast<char*>(&(RxBuffer[PacketStart + Packet.PayloadOffset()]));
207 
208  //call the actual command
209  //~ Cmds[Cmd].Response(Cmds[Cmd].Name, Params, Packet.PayloadLen(RxBuffer, RxCount, PacketStart), Argument);
210  Cmds[i].Response(Cmds[i].Name, Params, Packet.PayloadLen(RxBuffer, RxCount, PacketStart), (void*)this);
211  CmdFound = true;
212 
213  Processed = true;
214  }
215  }
216 
217  //Did we do anything?
218  if (!CmdFound)
219  {
220  //~ if (debug) { ::formatf("\n\nBinaryUart: Unmatched command 0x%.8lX!\n", PacketHeader->PayloadTypeToken); }
221  if (debug) { ::formatf("\n\nBinaryUart: Unmatched command 0x%.8lX! NumCmds: %lu\n", Packet.PayloadType(RxBuffer, PacketStart, PacketEnd), (unsigned long)NumCmds); }
222 
223  Callbacks.UnHandledPacket(reinterpret_cast<IPacket*>(&RxBuffer[PacketStart]), PacketEnd - PacketStart);
224  }
225  else
226  {
227  if (debug)
228  {
229  //~ ::formatf("\n\nBinaryUart: Got Packet! \n");
230  //~ PacketHeader->formatf();
231  //~ ::formatf(" <");
232  //~ for (size_t k = PacketStart; k < PacketEnd; k++)
233  //~ {
234  //~ ::formatf(":%0.2X", RxBuffer[k]);
235  //~ }
236  //~ ::formatf(">\n\n\n");
237  }
238  }
239  }
240  else
241  {
242  if (debug) { ::formatf("\n\nBinaryUart: Packet received, but SerialNumber comparison failed (expected: 0x%.8lX; got: 0x%.8lX).\n", SerialNum, Packet.SerialNum(RxBuffer, PacketStart, PacketEnd)); }
243 
244  Callbacks.UnHandledPacket(reinterpret_cast<IPacket*>(&RxBuffer[PacketStart]), PacketEnd - PacketStart);
245  }
246 
247  //Now just let the user do whatever they want with it...
248  Callbacks.EveryPacket(reinterpret_cast<IPacket*>(&RxBuffer[PacketStart]), PacketEnd - PacketStart);
249  }
250  else
251  {
252  if (debug) { ::formatf("\n\nBinaryUart: Packet received, but invalid.\n"); }
253 
254  Callbacks.InvalidPacket(reinterpret_cast<uint8_t*>(RxBuffer), RxCount);
255  }
256 
257  //~ ::formatf("\n\nBinaryUart: PacketEnd. (%u, %u): ", PacketEnd, RxCount);
258  //~ for(size_t i = PacketEnd - 4; i < RxCount; i++) { printf("%.2X:", RxBuffer[i]); }
259  //~ printf("\n\n");
260 
261  InPacket = false;
262 
263  if (RxCount > (PacketEnd + 4) )
264  {
265  size_t pos = 0;
266  size_t clr = 0;
267  for (; pos < (RxCount - (PacketEnd + 4)); pos++)
268  {
269  RxBuffer[pos] = RxBuffer[(PacketEnd + 4) + pos];
270  }
271  for (clr = pos; clr < RxCount; clr++)
272  {
273  RxBuffer[clr] = 0;
274  }
275  RxCount = pos;
276  //~ ::formatf("\n\nBinary Uart: PacketEnd. (%u, %u, %u): ", RxCount, (PacketEnd + 4), clr);
277  }
278  else
279  {
280  RxCount = 0;
281  RxBuffer[0] = 0;
282  }
283  }
284  else
285  {
286  if ( ((Packet.PayloadLen(RxBuffer, RxCount, PacketStart)) > RxBufferLenBytes) || ((Packet.PayloadLen(RxBuffer, RxCount, PacketStart)) > Packet.MaxPayloadLength()) )
287  {
288  //~ if (debug) { ::formatf("\n\nBinaryUart: Short packet (%lu bytes) with unrealistic payload len; ignoring corrupted packet (should have been header(%lu) + payload(%lu) + footer(%lu).\n", RxCount, sizeof(BinaryPacketHeader), PacketHeader->PayloadLen, sizeof(BinaryPacketFooter)); }
289  if (debug) { ::formatf("\n\nBinaryUart: Short packet (%lu bytes) with unrealistic payload len; ignoring corrupted packet (should have been header() + payload(%lu) + footer().\n", RxCount, Packet.PayloadLen(RxBuffer, RxCount, PacketStart)); }
290 
291  Callbacks.InvalidPacket(reinterpret_cast<uint8_t*>(RxBuffer), RxCount);
292 
293  InPacket = false;
294  RxCount = 0;
295  RxBuffer[0] = 0;
296  }
297  else
298  {
299  if (debug) { ::formatf("\n\nBinaryUart: Short packet (%lu bytes); we'll assume the packet footer was part of the payload data and keep searching for the packet end (should have been header() + payload(%lu) + footer().\n", RxCount, Packet.PayloadLen(RxBuffer, RxCount, PacketStart)); }
300  }
301  }
302  }
303  else { if (debug) { ::formatf("\n\nBinaryUart: Still waiting for packet end...\n"); } }
304  }
305 
306  return(Processed);
307  }
