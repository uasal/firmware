197  void Init()
198  {
199  for (uint8_t i = 0; i < 10; i++)
200  {
201  uint8_t AdcInit = Adc.Init(GpioDirections);
202  if (Adc.InitOK == AdcInit)

205  break;
206  }
207  else
208  {
209  formatf("\nMonitorAdc: Error initializing Adc: %u [0x%.2X].\n", AdcInit, AdcInit);
210  }

249  void Process()
250  {
251  if (AdcFound)
252  {
253  //~ we don't have anything connected to gpios but might in the future, and dm does if this code gets cut&pasted: GpioInputs = Adc.ReadRegister(ads1258details::register_gpiod);
254 
255  ads1258details::ads1258sample sample;
256 
257  for(size_t CurrentChan = 0; CurrentChan < ads1258details::ads1258numchannels; CurrentChan++)
258  {
259  ProcessAllUarts();
260 
261  Adc.AutoScan();
262 
263  if (Adc.IsScanChannel(CurrentChan))
264  {
265  ProcessAllUarts();
266 
267  Adc.GetLastSample(CurrentChan, sample);
268 
269  ProcessAllUarts();
270 
271  if ( (sample.status.isbrownout) || (sample.status.isclipped) )
272  {
273  if (Monitor) { ::formatf("\nMonitorAdc: ch %u bad status: 0x%.2X\n", sample.status.channel, sample.status.all); }
274  }
275  else
276  {
277  if (sample.status.isnew)
278  {
279  if (Monitor) { ::formatf("\nMonitorAdc: ch %u reads: %lf Volts [%lu lsb's]\n", sample.status.channel, Adc.CountsToVolts(sample.sample), sample.sample); }
280 
281  switch(sample.status.channel)
282  {
283  case P1V2Channel : { P1V2 = (sample.sample); break; }
284  case P2V2Channel : { P2V2 = (sample.sample); break; }
285  case P28VChannel : { P28V = (sample.sample); break; }
286  case P2V5Channel : { P2V5 = (sample.sample); break; }
287  case P6VChannel : { P6V = (sample.sample); break; }
288  case P5VChannel : { P5V = (sample.sample); break; }
289  case P3V3DChannel : { P3V3D = (sample.sample); break; }
290  case P4V3Channel : { P4V3 = (sample.sample); break; }
291  case P2I2Channel : { P2I2 = (sample.sample); break; }
292  case P4I3Channel : { P4I3 = (sample.sample); break; }
293  case P6IChannel : { P6I = (sample.sample); break; }
294  case Aux0Channel : { Aux0 = (sample.sample); break; }
295  case Aux1Channel : { Aux1 = (sample.sample); break; }
296  case Aux2Channel : { Aux2 = (sample.sample); break; }
297  case AmbientLightChannel : { AmbientLight = (sample.sample); break; }
298  case TemperatureChannel : { Temperature = (sample.sample); break; }
299 
300  default: { break; } //do nothing; we can scan all 16 channels and discard the unused ones.
301  }
302  }
303  else
304  {
305  if (Monitor) { ::formatf("\nMonitorAdc: ch %u no new sample this scan.\n", sample.status.channel); }
306  }
307  }
308  }
309  }
310  }
