379 
380  size_t len = BufferLen;
381  if (len > 32) { len = 32; }
382  formatf("\nFPGAUartCallbacks: InvalidPacket! contents: :");
383  for(size_t i = 0; i < len; i++) { formatf("%.2X:", Buffer[i]); }
384  formatf("\n\n");
385  }

387  //Packet with no matching command handler:
388  virtual void UnHandledPacket(const IPacket* Packet, const size_t& PacketLen)
389  {
390  if ( (NULL == Packet) || (PacketLen < sizeof(CGraphPacketHeader)) ) { formatf("\nFPGABinaryUartCallbacks: NULL(%u) UnHandledPacket!\n\n", PacketLen); return; }
391 
392  const CGraphPacketHeader* Header = reinterpret_cast<const CGraphPacketHeader*>(Packet);
393  formatf("\nFPGAUartCallbacks: Unhandled packet(%u): ", PacketLen);

395  formatf("\n\n");
