155 void ParseConfigFile(const char* ConfigFileName)

163  if (((int)ParseFileHandle) < 0)
164  {
165  perror("\nMapParseConfigFileDataBuffer(): error in open(): ");
166  return;
167  }
168  else
169  {
170  struct stat s;
171  //~ int err = stat(ConfigFileName, &s);
172  int err = fstat(ParseFileHandle, &s);
173  if (0 != err)
174  {
175  perror("\nMapParseConfigFileDataBuffer(): error in stat(): ");
176  return;
177  }
178 
179  FileLen = s.st_size;
180 
181  ParseConfigFileDataBuffer = (uint8_t*)mmap(0, FileLen, PROT_READ, MAP_SHARED, (int)ParseFileHandle, 0);
182  if (MAP_FAILED == ParseConfigFileDataBuffer)
183  {
184  ParseConfigFileDataBuffer = NULL;
185  printf("\nMapParseConfigFileDataBuffer(%zu): error in mmap(%d): ", FileLen, errno);
186  return;
187  }
188  else
189  {
190  printf("\nMapParseConfigFileDataBuffer(): Mapped at: %p; len: %zu.\n", ParseConfigFileDataBuffer, FileLen);
191  }
192  }
193 
194  printf("\n\nStartup configuration file %s opened, handle: %d, preparing to parse commands...\n", ConfigFileName, ParseFileHandle);
195 
196  //Start finding lines
197  for (BufPos = 0; BufPos < FileLen; BufPos++)
198  {
199  //Find the end of the next line in the buffer
200  uint8_t* c = (uint8_t*)memchr(&(ParseConfigFileDataBuffer[BufPos]), '\n', FileLen - BufPos);
201 
202  //~ printf("\nParseConfigFile(): found '\\n' in buffer at %lu.\n\n", (uint32_t)(c - snprintfBuffer));
203 
204  //done with this buffer:
205  if (NULL == c) { break; }
206 
207  //Move into seperate buffer so parser can't muck us up...
208  size_t LineLen = c - &(ParseConfigFileDataBuffer[BufPos]);
209  if (LineLen > 4095) { LineLen = 4095; }
210  memcpy(Line, &(ParseConfigFileDataBuffer[BufPos]), LineLen);
211  Line[LineLen] = '\n';
212  Line[LineLen + 1] = '\0';

222  }

228  }
