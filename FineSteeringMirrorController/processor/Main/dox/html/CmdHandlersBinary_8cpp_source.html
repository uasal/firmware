203 int8_t BinaryFWFilterSelectCommand(const uint32_t Name, char const* Params, const size_t ParamsLen, const void* Argument)
204 {
205  uint32_t FilterSelect = 0;
206 
207  formatf("\nBinaryFWFilterSelectCommand: processing(%u)...\n\n", ParamsLen);
208 
209  if ( (NULL != Params) && (ParamsLen >= (sizeof(uint32_t))) )
210  {
211  FilterSelect = *(const uint32_t*)Params;
212 
213  if (FilterSelect > FWMaxPosition)
214  {
215  formatf("\nBinaryFWFilterSelectCommand: Invalid position requested: %lu; max valid position: %lu; assuming we've been requested to re-home wheel\n", FilterSelect, FWMaxPosition);
216  FWHome();
217  FilterSelect = (uint32_t)-1; //-1 means we're in motion
218  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &FilterSelect, sizeof(uint32_t));
219  return(ParamsLen);
220  }
221 
222  uint32_t response = (uint32_t)-1; //-1 means we're in motion
223  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &response, sizeof(uint32_t));
224  formatf("\nBinaryFWFilterSelectCommand: moving to: %lu\n", FilterSelect);
225  FWSeekPosition(FilterSelect);
226  if (!ValidateFWPostition())
227  {
228  response = (uint32_t)-2; //-2 means invalid position
229  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &response, sizeof(uint32_t));
230  formatf("\n\nBinaryFWFilterSelectCommand: move failed!\n");
231  FWHome();
232  formatf("\n\nBinaryFWFilterSelectCommand: moving to: %lu\n", FilterSelect);
233  FWSeekPosition(FilterSelect);
234  //sure hope it worked the second time, cause I'm not sure we have the werewithal to try recursively...
235  }
236  response = (uint32_t)FilterSelect; //-1 means we're in motion
237  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &response, sizeof(uint32_t));
238  return(ParamsLen);
239  }
240 
241  //if we get here they must've wanted to know where we are
242  formatf("\n\nBinaryFWFilterSelectCommand: querying current position...\n");
243 
244  //Is motor moving?
245  CGraphFWMotorControlStatusRegister MCSR;
246  MCSR = FW->MotorControlStatus;
247  if (MCSR.SeekStep != MCSR.CurrentStep)
248  {
249  formatf("\n\nBinaryFWFilterSelectCommand: motor is in motion to target position: %lu\n", FWPosition);
250  FilterSelect = (uint32_t)-1; //-1 means we're in motion
251  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &FilterSelect, sizeof(uint32_t));
252  return(ParamsLen);
253  }
254 
255  //If not, do we have any idea where we are?
256  if (!ValidateFWPostition())
257  {
258  formatf("\n\nBinaryFWFilterSelectCommand: current position invalid!!\n");
259  FilterSelect = (uint32_t)-2; //-2 means we have no idea where we are!
260  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &FilterSelect, sizeof(uint32_t));
261  FWHome();
262  return(ParamsLen);
263  }
264 
265  //If we get here we are where we say we are, and it's all shiny, cap'n!
266  formatf("\n\nBinaryFWFilterSelectCommand: current position: %lu\n", FWPosition);
267  FilterSelect = FWPosition;
268  TxBinaryPacket(Argument, CGraphPayloadTypeFWFilterSelect, 0, &FilterSelect, sizeof(uint32_t));
269 
270  return(ParamsLen);
271 }
